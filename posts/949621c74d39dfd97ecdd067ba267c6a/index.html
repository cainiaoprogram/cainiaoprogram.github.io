【next】nextjs打包后出现passHref is missing

解决流程

import Link from 'next/link'


const NavLink = ({ href, name }) => {
  return (
  //添加passHref
    <Link href={href} passHref>
      {name}
    </Link>
  )
}
export default NavLink
