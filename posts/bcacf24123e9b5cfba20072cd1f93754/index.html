oracle 修改表字段顺序

第一步 查询表ID

select object_id
from all_objects
where owner = '当前用户名'
and object_name = '表名';

第二步 查询表字段序号

select obj#, col#, name
from sys.col$
where obj# = '第一步的object_id' order by col#;

第三步 修改字段顺序

update sys.col$ set col#=3 where obj#='第一步的object_id' and name='要修改的字段名称';
