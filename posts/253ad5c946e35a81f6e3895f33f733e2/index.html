孪生素数C语言

题目描述

编程输入正整数a(10000>=a>=2)，输出不小于a的第一对孪生素数。差是2的两个素数被称为孪生素数。

输入

无

输出

无

输入样例

10000

输出样例

10007 10009

代码演示

#include <stdio.h>
#include <stdlib.h>
int sushu(int i)
 
{
    int j,flag;
    if(i==1)
        flag=0;
    for(j=2; j*j<=i; j++)
    {
        if(i%j==0)
        {
            flag=1;
            break;
        }
        else
 
        {
            flag=0;
            continue;
        }
    }
    return flag;
}
 
int main()
 
{
    int a,n;
    scanf("%d",&a);
    {
        for(n=a; ; n++)
 
        {
            if(sushu(n)==0&&sushu(n+2)==0)
                break;
            else
                continue;
        }
        printf("%d %d",n,n+2);
 
    }
    return 0;
}
