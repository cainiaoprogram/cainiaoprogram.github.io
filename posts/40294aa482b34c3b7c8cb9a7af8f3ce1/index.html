118. 杨辉三角

给定一个非负整数 numRows，生成杨辉三角的前 numRows 行。

在杨辉三角中，每个数是它左上方和右上方的数的和。

示例:

输入: 5
输出:
[
     [1],
    [1,1],
   [1,2,1],
  [1,3,3,1],
 [1,4,6,4,1]
]
通过次数138,732提交次数199,577

/**
 * @param {number} numRows
 * @return {number[][]}
 */
var generate = function(numRows) {

    const res = [];

    for(let i=0;i<numRows;i++) {
        const row = [];
        for(let j=0;j<=i;j++) {
            if(j === 0) {
                row.push(1);
            }
            else if(i === j) {
                row.push(1);
            }
            else {
                row.push(res[i-1][j-1] + res[i-1][j]);
            }
        }
        res.push(row);
    }

    return res;
};
