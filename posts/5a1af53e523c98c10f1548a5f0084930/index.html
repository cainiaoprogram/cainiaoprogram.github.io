operater int()

class Number
{
    int number;
public:
    explicit Number(int n){number = n;}
    operator int() //注意一定不能声明返回值
    {
        return number;
    }
};
int main () {

    Number n1 = Number(100);
    int n2= n1;
    cout << n2 << endl;
    return 0;
}
