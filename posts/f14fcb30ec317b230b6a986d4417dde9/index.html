 2.年月拼接，pre,cur,next

let date = new Date();
      let year = date.getUTCFullYear()
      let month = date.getUTCMonth() + 1
      let pre = '';
      let cur = '';
      let next = '';
      cur = year + '-' + (month < 9 ? ('0' + month) : month)
      pre = month === 1 ? ((year - 1) + '-12') : (year + '-' + (month - 1 < 9 ? ('0' + (month - 1)) : month - 1))
      next = month === 12 ? ((year + 1) + '-01') : (year + '-' + (month + 1 > 9 ? ('0' + (month + 1)) : month + 1))
