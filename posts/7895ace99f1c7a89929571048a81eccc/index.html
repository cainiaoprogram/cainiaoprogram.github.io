C# – Mysql 数据库连接
1. 配置环境

#前提：电脑已安装Mysql服务；
Visual Studio 安装Mysql依赖库：
    工具 -> NuGet 包管理器 -> 管理解决方案的 NuGet程序包 —> 搜索， 安装Mysql.Data (Oracle); (安装成功后，在项目引用里面可以看到 Mysql.Data)
 

接着安装 System.Data.SqlClient

这样就安装完成了，接下来就是代码了

using MySql.Data.MySqlClient;
using System;
using System.Collections.Generic;
using System.ComponentModel;
using System.Data;
using System.Drawing;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Windows.Forms;

namespace MySqlWinForm
{
    public partial class MySqlForm : Form
    {
        public MySqlForm()
        {
            InitializeComponent();
        }

        private void btnMySqlConnect_Click(object sender, EventArgs e)
        {

            string connectUrl = "server = localhost; " + //ip地址 本机(localhost | 127.0.0.1) 
                "port = 3306; " + //端口号 mysql 默认端口(3306)
                "user = root; " + //用户名
                "password = a6; " + //密码
                "database = mysql"; //数据库名称
            MySqlConnection conn = new MySqlConnection(connectUrl);
            try
            {
                //可能出现异常
                conn.Open();
                MessageBox.Show("连接成功!");
            }
            catch (MySqlException ex)
            {
                //异常则提示异常信息
                MessageBox.Show(ex.Message);
            }
            finally
            {
                //务必关闭 MysqlConnection
                conn.Close();
            }



        }

    }
}
