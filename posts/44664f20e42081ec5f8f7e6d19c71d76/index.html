获取输入日期的下一天(Java)

 年份的范围在1920-2050之间

代码实现 

import java.util.Scanner;

//获取输入日期的下一天
public class NextDate {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        NextDate nextdate = new NextDate();

        System.out.println("请输入年份:");
        int year = scanner.nextInt();
        System.out.println("请输入月份:");
        int month = scanner.nextInt();
        System.out.println("请输入天数:");
        int day = scanner.nextInt();

        nextdate.DataWay(year,month,day);
    }
    private int year;//年
    private int month;//月
    private int day;//日
    //获取输入日期的下一天
    public void DataWay(int year,int month,int day){
        this.year = year;
        this.month = month;
        this.day = day;
        //检验天数
        if(!(day >= 1 && day <= 31)){
            System.out.println(day+"日为无效等价类，应在1-31之间");
            return;
        }
        //检验年份,1920 <= year <= 2050
        if(year >= 1920 && year <= 2050) {
            //检测月份,大年1 3 5 7 8 10 12
            if (month == 1 || month == 3 || month == 5 || month == 7 || month == 8 || month == 10 || month == 12) {
                if (this.day >= 1 && this.day <= 30) {
                    this.day += 1;
                }
                if (this.day == 31) {
                    this.month += 1;
                    this.day = 1;
                    if(this.month >= 12){
                        this.year += 1;
                        this.month = 1;
                        if(this.year > 2050){
                            System.out.println(this.year+"年应在1920-2050之内!");
                            return;
                        }
                    }
                }
                System.out.println(this.year+"年"+this.month+"月"+this.day+"日");
            }
            //检测小年4 6 9 11
            else if (month == 4 || month == 6 || month == 9 || month == 11) {
                if (this.day >= 1 && this.day <= 29) {
                    this.day += 1;
                }
                if (this.day == 30) {
                    this.month += 1;
                    this.day = 1;
                    if(this.month >= 12){
                        this.year += 1;
                        this.month = 1;
                        if(this.year > 2050){
                            System.out.println(this.year+"年应在1920-2050之内!");
                            return;
                        }
                    }
                }
                System.out.println(this.month+"月"+this.day+"日");
            }
            //月份为2月
            else if (month == 2) {
                if ((year % 400 == 0) || (year % 4 == 0 && year % 100 != 0)) {
                    if (this.day >= 1 && this.day <= 27) {
                        this.day += 1;
                    }
                    if (this.day == 28) {
                        this.month += 1;
                        this.day = 1;
                        if(this.month >= 12){
                            this.year += 1;
                            this.month = 1;
                            if(this.year > 2050){
                                System.out.println(this.year+"年应在1920-2050之内!");
                                return;
                            }
                        }
                    }
                    System.out.println(this.month+"月"+this.day+"日");
                } else {
                    this.month = month;//平年2月29天
                    if (this.day >= 1 && this.day <= 28) {
                        this.day += 1;
                    }
                    if (this.day == 29) {
                        this.month += 1;
                        this.day = 1;
                        if(this.month >= 12){
                            this.year += 1;
                            this.month = 1;
                            if(this.year > 2050){
                                System.out.println(this.year+"年应在1920-2050之内!");
                                return;
                            }
                        }
                    }
                    System.out.println(this.month+"月"+this.day+"日");
                }
            }
            else{
                System.out.println(this.month+"月应在1-12之内!");
            }
        }
    }
}
