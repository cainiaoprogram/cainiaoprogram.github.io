js

function uploadImg(file){
            if (file.files && file.files[0])
            {
                var img = document.getElementById('img');
                var reader = new FileReader();
                reader.readAsDataURL(file.files[0]);
                reader.onloadend = function(e){
                    $('#img-prize').show();
                    img.src = e.target.result;
                }

            }
        }
