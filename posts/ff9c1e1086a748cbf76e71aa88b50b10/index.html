题目：最小栈

  • 题目描述

设计一个支持 push，pop，top 操作，并能在常数时间内检索到最小元素的栈。

  • push(x) – 将元素x推入栈中。
  • pop() – 删除栈顶的元素。
  • top() – 获取栈顶元素。
  • getMin() – 检索栈中的最小元素。

示例：

MinStack minStack = new MinStack();
minStack.push(-2);
minStack.push(0);
minStack.push(-3);
minStack.getMin(); –> 返回 -3.
minStack.pop();
minStack.top(); –> 返回 0.
minStack.getMin(); –> 返回 -2.

#include <stdio.h>
#include <stack>
class MinStack {
public:
    /** initialize your data structure here. */
    MinStack() {

    }

    void push(int x) {    //将数据压入数据栈
        _data.push(x);  //将数据压入数据栈
        if(_min.empty())   //如果
        {
            _min.push(x);
        }
        else{
            if(x>_min.top())
            {
               x = _min.top();
            }
            _min.push(x);
        }
    }

    void pop() {   //数据栈与最小值栈同时弹出
        _data.pop();
        _min.pop();
    }

    int top() {  //top即返回数据栈顶
        return _data.top();
    }

    int getMin() {  //getMin即返回最小值栈顶
        return _min.top();
    }
private:
    std::stack<int> _data;  //数据栈
    std::stack<int> _min;  //最小值栈
};

 * int param_4 = obj.getMin();
 */
