6-1 使用函数求素数和（20 分）

本题要求实现一个判断素数的简单函数、以及利用该函数计算给定区间内素数和的函数。

素数就是只能被1和自身整除的正整数。注意：1不是素数，2是素数。

函数接口定义：

int prime( int p );
int PrimeSum( int m, int n );

其中函数prime当用户传入参数p为素数时返回1，否则返回0；函数PrimeSum返回区间[m, n]内所有素数的和。题目保证用户传入的参数m≤n。

裁判测试程序样例：

#include <stdio.h>
#include <math.h>

int prime( int p );
int PrimeSum( int m, int n );

int main()
{
    int m, n, p;

    scanf("%d %d", &m, &n);
    printf("Sum of ( ");
    for( p=m; p<=n; p++ ) {
        if( prime(p) != 0 )
            printf("%d ", p);
    }
    printf(") = %d\n", PrimeSum(m, n));

    return 0;
}
int prime(int p)
{ if(p<=1) return 0;
  if(p==2) return 1;
  int i;
  for(i=2;i<=sqrt(p);i++)
    if(p%i==0) return 0;
  return 1;
}
int PrimeSum(int m,int n)
{
  int i,sum=0;
  for(i=m;i<=n;i++)
   if(prime(i)==1) sum +=i;
  return sum;
}
6-2 使用函数验证哥德巴赫猜想（20 分）

函数接口定义：

int prime( int p );
void Goldbach( int n );

其中函数prime当用户传入参数p为素数时返回1，否则返回0；函数Goldbach按照格式“n=p+q”输出n的素数分解，其中p≤q均为素数。又因为这样的分解不唯一（例如24可以分解为5+19，还可以分解为7+17），要求必须输出所有解中p最小的解。

裁判测试程序样例：

#include <stdio.h>
#include <math.h>

int prime( int p );
void Goldbach( int n );

int main()
{
    int m, n, i, cnt;

    scanf("%d %d", &m, &n);
    if ( prime(m) != 0 ) printf("%d is a prime number\n", m);
    if ( m < 6 ) m = 6;
    if ( m%2 ) m++;
    cnt = 0;
    for( i=m; i<=n; i+=2 ) {
        Goldbach(i);
        cnt++;
        if ( cnt%5 ) printf(", ");
        else printf("\n");
    }

    return 0;
}
int prime(int p)
{ if(p<=1) return 0;
  if(p==2) return 1;
  int i;
  for(i=2;i<=sqrt(p);i++)
    if(p%i==0) return 0;
  return 1;
}
void Goldbach( int n )
{ int i = n;
  while(prime(i)!=1||prime(n-i)!=1) i--;
  printf("%d=%d+%d",n,n-i,i);
}
