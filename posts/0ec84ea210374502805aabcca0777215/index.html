PAT 乙级 1013 数素数

解题思路，简单素数判定，计算数量 输出就可以了

#include<stdio.h>
int prime(int n)
{
	int i,flag=1;
	for(i=2;i*i<=n;i++)
	{
		if(n%i==0)
		{
			flag=0;
			break;
		}
	}
	return flag;
	
}
int main()
{
	int n,m,amont=2,count=0,space=0;
	scanf("%d %d",&n,&m);
	while(count<n-1)
	{
		if(prime(amont))
		{
			
			count++;
		}
		amont++;
	}
	
	while(count<m)
	{
		if(prime(amont))
		{
		    
			count++;
			space++;
			if(space%10!=0&&count!=m)
			printf("%d ",amont);
			else
			printf("%d\n",amont);
			
		}
		amont++;
	}
	return 0;
}
