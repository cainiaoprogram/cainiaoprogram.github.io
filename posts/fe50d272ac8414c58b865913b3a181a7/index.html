C语言：输入10个数整数，输出最大（或最小）的数。

#include <stdio.h>
int main(void)
{
int a[10],i,max = 0;
printf(“请输入十个整数：”);
for(i = 0;i < 10;i++)
scanf("%d",&a[i]);
for(i = 0;i < 10;i++)
if(a[i] > a[max])
max = i;
printf(“最大的整数为：%d\n”,a[10]);
return 0;
}
