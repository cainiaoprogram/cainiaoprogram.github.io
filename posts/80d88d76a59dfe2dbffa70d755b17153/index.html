1.获取Token 再发送Get请求

#Post请求，获取Token
def get_token():
    url_login = r'http://xxx:8080/api/login'
    headers_login = { 'Content-Type': 'application/json;charset=UTF-8'}
    data = {'user': 'abc', 'password': '123456'}
    j_data = json.dumps(data)
    resp = requests.post(url=url_login, headers=headers_login, data=j_data)
    return resp.json()['token']

#get请求
def set_data():
    url_set = r'http://xxx:8080/api/setdata'
    headers = {'X-Token': get_token()}    #通过上面的get_token()方法，获得Token
    para_data={'max': '100', 'min': '50'}
    resp = requests.get(url=url_set, headers=headers, params=para_data)
    #print (resp.text)
    return resp.status_code
