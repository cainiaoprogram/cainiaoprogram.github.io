package lyy;

public class oprator {
	private float first;
	private float second;
	private String op;
	private float result;
	
	public float getFirst() {
		return first;
	}
	public void setFirst(float first) {
		this.first = first;
	}
	public float getSecond() {
		return second;
	}
	public void setSecond(float second) {
		this.second = second;
	}
	
	
	public String getOp() {
		return op;
	}
	public void setOp(String op) {
		this.op = op;
	}
	public float getResult() {
		if (op.equals("+")) {
			result = first+second;
		}
		if (op.equals("-")) {
			result = first-second;
		}
		if (op.equals("*")) {
			result = first*second;
		}
		if (op.equals("/") && second != 0) {
			result = first/second;
		}
		return result;
	}
	public void setResult(float result) {
		this.result = result;
	}
	

}
