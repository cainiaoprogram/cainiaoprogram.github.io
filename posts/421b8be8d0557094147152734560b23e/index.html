蓝桥杯备考日记0202特殊数的和

题目如下：

解题过程：

#include<iostream>
using namespace std;
int main()
{
    int n;
    int sum = 0;
    cin >> n;
    for(int i = 1;i <= n;i++)
    {
        int temp=i;//定义临时变量，方便后续判断i的各个位上的数是否是特殊的数 
        while(temp)
        {
            if((temp%10)==2||(temp%10)==0||(temp%10)==1||(temp%10)==9)
            {
                sum=sum+i;
                break;
            }
            else  temp=temp/10;
        }
    }
    cout<<sum<<endl; 
    return 0;
 }
