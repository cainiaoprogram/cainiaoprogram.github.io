MFC设置编辑框背景颜色等

1、在Dlg.h头文件中添加函数声明

添加

public:
	HBRUSH OnCtlColor(CDC * pDC, CWnd * pWnd, UINT nCtlColor);
2、重载OnCtlColor(CDC* pDC, CWnd* pWnd, UINT nCtlColor)

即添加ON_WM_CTLCOLOR()

3、编辑OnCtlColor(CDC* pDC, CWnd* pWnd, UINT nCtlColor)函数
3.1 对单个控件操作
HBRUSH C菜单打钩Dlg::OnCtlColor(CDC* pDC, CWnd* pWnd, UINT nCtlColor)
{
	HBRUSH hbr = CDialog::OnCtlColor(pDC, pWnd, nCtlColor);
	// TODO: 在此添加控件通知处理程序代码
	if (pWnd->GetDlgCtrlID() == IDC_MESG)		//如果是ID为IDC_MESG的编辑框
	{
		pDC->SetTextColor(RGB(255, 0, 0));   //设置字体颜色
		pDC->SetBkMode(TRANSPARENT); 		//设置字体背景为透明
									 // TODO: Return a different brush if the default is not desired
		return (HBRUSH)::GetStockObject(BLACK_BRUSH);   // 设置背景色
	}
	else
		return hbr;
}
3.2 对对话框所有控件操作

在头文件中声明

public:
	CBrush m_Brush;  //声明画刷

在OnInitDialog()中添加

	m_Brush.CreateSolidBrush(RGB(166, 243, 248));		//创建蓝色的画刷

HBRUSH C菜单打钩Dlg::OnCtlColor(CDC* pDC, CWnd* pWnd, UINT nCtlColor)
{
	HBRUSH hbr = CDialog::OnCtlColor(pDC, pWnd, nCtlColor);
	// TODO: 在此添加控件通知处理程序代码
	if (nCtlColor == CTLCOLOR_DLG)
		return m_Brush;
	if (nCtlColor == CTLCOLOR_STATIC)		//所有静态控件颜色
	{
		pDC->SetTextColor(RGB(0, 0, 0));		//所有静态文本颜色（黑色）
		pDC->SetBkColor(RGB(166, 243, 248));	//背景色
		hbr = (HBRUSH)m_Brush;
	}
	if (nCtlColor == CTLCOLOR_EDIT)			//所有编辑框颜色
	{
		pDC->SetTextColor(RGB(0, 0, 0));
		pDC->SetBkMode(TRANSPARENT); 		//设置背景模式为透明
		hbr = (HBRUSH)m_Brush;
	}
	return hbr;
}
