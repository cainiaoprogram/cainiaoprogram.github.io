30：C语言之亲密数

#include<stdio.h>
void solve(){
/*********Begin*********/
	int a,b,c=0,m=0;
	for(a=1;a<=3000;a++)
	{	
		for(b=1;b<a;b++)
		{
			if(a%b==0)
				c=c+b;
		}
		for(b=1;b<c;b++)
		{
			if(c%b==0)
				m=m+b;
		}
		if(m==a)
		{
			if(a!=c&&a<c)
			printf("(%d,%d)",a,c);
		}
		c=0,m=0;
		
	}
    /*********End**********/ 
}
int main(void)
{  
    solve();
    return 0;
}
