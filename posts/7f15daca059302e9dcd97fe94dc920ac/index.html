// 钉钉免登

					let that = this
					dd.ready(function() {
						dd.runtime.permission.requestAuthCode({
							corpId:'钉钉公司的corpid',
							onSuccess: function(result) {
								console.log(result);
								that.authCode = result.code;
								that.$http.postjson("/ding/login",{code:result.code}).then((res) => {

									uni.setStorageSync('Authorization',res.data);
								});
							},

								console.log(err)
							}
