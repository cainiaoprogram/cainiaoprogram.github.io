80C51 单片机 继电器控制灯的亮灭

#include<reg52.h>
#include<intrins.h>
sbit light=P2^4;		//将light设置为P2的第4位,该IO口连接继电器。
#define uchar unsigned char
#define uint unsigned int
void delay(uint ms);
void main()
{
	while(1)
	{
		light=0;	//灯灭
		delay(200);
		light=1;	//灯亮
		delay(300);
	}
}

void delay(uint ms)	   //延时函数
{
	uint i,j;
	for(i=ms;i>0;i--)
	for(j=200;j>0;j--);
}
