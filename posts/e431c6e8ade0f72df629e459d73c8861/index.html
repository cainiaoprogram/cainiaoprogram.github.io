集合set

# set集合
set1 = {1, 2, 3, 4, 5, 6}
set2 = {4, 5, 6}
# set1.add(7)
# set1.discard(3)
print(set1)
# 取交集
print(set1.intersection(set2))
# 取没有的集合
print(set1.difference(set2))
# 判断是否子集
print(set2.issubset(set1))
