P1952 火星上的加法运算

#include <bits/stdc++.h>
using namespace std;

const int maxn = 2010;
const char* output = "0123456789abcdefghijklmnopqrstuvwxyz";
char s1[maxn],s2[maxn];
int a[maxn],b[maxn],c[maxn];

int change(char ch)
{
    if(isalpha(ch)) return ch - 'a' + 10;
    else return ch - '0';
}
int main()
{
    int mod;
    scanf("%d",&mod);
    scanf("%s%s",s1,s2);
    int l1 = strlen(s1),l2 = strlen(s2);
    for(int i = 0;i < l1;i++) a[i] = change(s1[l1 - i - 1]);
    for(int i = 0;i < l2;i++) b[i] = change(s2[l2 - i - 1]);
    int l3 = max(l1,l2);
    for(int i = 0;i < l3;i++)
    {
        c[i] += a[i] + b[i];
        c[i + 1] += c[i]/mod;
        c[i] %= mod;
    }
    if(c[l3]) l3++;
    for(int i = l3 - 1;i >= 0;i--)  printf("%c",output[c[i]]);
    return 0;
}
