首先，为了进行验证，我编了一个简单的程序test.exe(在C盘系统文件中创建一个文件)：

#include <iostream>
#include <fstream>

using namespace std;
int main()
{
    ofstream temp("C:/Program Files (x86)/Internet Explorer/test/test.txt");

    if (!temp)
        cout << "creat failed" << endl;
    else
    {
        cout << "creat successfully" << endl;
        temp.close();
    }
    system("pause");
    return 0;
}
