题目三：实现斐波那契（Fibonacci）数列

方法一：递归

def fibonacci2(n):
    if n == 1 or n == 2:
        return 1
    else:
        return fibonacci2(n-1) + fibonacci2(n-2)


方法二：迭代

def fibonacci(n):
    if n == 1 or n == 2:
        return 1

    nPre = 1
    nLast = 1
    nResult = 0
    i = 2
    while i < n:
        nResult = nPre + nLast
        nPre = nLast
        nLast = nResult
        i += 1

    return nResult

print fibonacci(5)
