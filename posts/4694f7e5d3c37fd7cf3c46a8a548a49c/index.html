按字段field4降序排序，按字段field1去重

SELECT tt1.name2,tt1.field1,tt1.field2,tt1.field4
from 
(
select tt2.name2,tt2.field1,tt2.field2,tt2.field4
from t2 tt2
ORDER BY tt2.field4 DESC
) tt1
GROUP BY tt1.field1
