def left_shift(based, counter):
    val = based << (counter % 32)
    maxint = 2147483647
    if not -maxint - 1 <= val <= maxint:
        val = (val + (maxint + 1)) % (2 * (maxint + 1)) - maxint - 1
    return val

import ctypes
def left_shift(x, y):
    x,y = ctypes.c_int32(x).value,y % 32
    return ctypes.c_int32(x << y).value
