zookeeper hub.docker官方镜像使用方法

#下载hub.docker官方镜像
docker pull zookeeper
#--restart always 自动重启
#--name zookeeper 设置容器名
#-p 2181:2181主机端口与容器端口映射，不设置无法使用
#-p 8081:8080主机端口8081与容器端口8080映射。本机8080端口已被其他服务占用。
docer run --restart always --name zookeeper -p 2181:2181 -p 8081:8080 -p zookeeper
