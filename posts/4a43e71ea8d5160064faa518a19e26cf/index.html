3.printf()函数的使用方法

main()

{

int a=1234;

int b=255;

float f=123.456;

char ch=‘a’;

printf(“%8d,%2d\n”,a,a);

/*输出10进制整数，字段宽度为8位；输出10进制整数，字段宽度2位*/

printf(“%d\t%o\t%x\n”,b,b,b);

/*输出整数，以8进制数输出，以16进制数输出 */

printf(“%f,%8f,%8.1f,%.2f,%.2e\n”,f,f,f,f,f);

/*输出浮点数，输出8位浮点数，输出8.1浮点数？，输出.2浮点数，以指数形式输出 */

printf(“%3c\n”,ch);

/*输出3位字符数 */

}
