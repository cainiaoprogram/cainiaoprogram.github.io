PAT乙级 1011 A+B 和 C

解题思路
本题非常简单但是一样有坑，c语言写的时候一定要注意范围，如果你只用int型去写的话最大值相加一定溢出，所以你必须扩大字节数量，改成long long int,

python没有此问题

c代码：

#include<stdio.h>
int main()
{
	int n,a,b,c,i;
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		scanf("%d %d %d",&a,&b,&c);
		if(a+b>c)
		printf("Case #%d: true\n",i+1);
		else
		printf("Case #%d: false\n",i+1);
	}
	return 0;
 }

python 代码

n=int(input())
for i in range(n):
    s=input().split()
    if(eval(s[0]+'+'+s[1])>eval(s[2])):
        print("Case #{}: true".format(i+1))
    else:
        print("Case #{}: false".format(i+1))
