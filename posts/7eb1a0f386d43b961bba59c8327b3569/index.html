                 yarn add qrcode.vue  -S

<template>
  <div>二维码生成</div>
  <qrcode-vue :value="data.payUrl" :size="data.size" level="H" />
</template>
 
<script  setup>
import { reactive, ref } from 'vue'
import QrcodeVue from 'qrcode.vue'
 
const data = reactive({
