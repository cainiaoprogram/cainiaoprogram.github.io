ios 使用webview播放视频需注意

videoWeb是个webView

#pragma mark UITableViewDelegate

- (void)tableView:(UITableView *)tableView didSelectRowAtIndexPath:(NSIndexPath *)indexPath{

    videoWeb = [[UIWebView alloc] initWithFrame:CGRectMake(10,0,3,2)];

    videoWeb.hidden =NO;

    videoWeb.delegate =self;

    [self.view addSubview:videoWeb];

    

 excellentModel.urlArr 里面是视频的url数组

    NSURL *url = [NSURL URLWithString:[excellentModel.urlArr  objectAtIndex:indexPath.row]];//创建URL

    NSURLRequest* request = [NSURLRequest requestWithURL:url];

    [videoWeb loadRequest:request];

}


#pragma mark UIWebDelegate

- (void)webViewDidFinishLoad:(UIWebView *)webView{

    videoWeb =nil;

}
