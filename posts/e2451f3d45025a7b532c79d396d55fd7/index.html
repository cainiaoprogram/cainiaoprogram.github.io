if(!function_exists('randInsertStr')){

    /**
     * 随机插入字符串
     * @param string $string 原字符串
     * @param string $insertStr 需要插入的字符串
     * @return bool|string
     * @author Lj
     * @date 2020-11-12
     */
    function randInsertStr($string){
        if(empty($string)){
            return false;
        }
        preg_match_all("/./u", $string, $arr);
        $arrNew = array(1,2);
        $str = '';
        foreach ($arr[0] as $k=>$v){
            if($k == 0)
            {
                $str.= $v.' ';
            }else{
                $key = array_rand($arrNew,1);
                $rand = $arrNew[$key];
                $str.= ($rand % 2) == 0 ?  $v.' ': $v;
            }
        }
        return $str;
    }
}
