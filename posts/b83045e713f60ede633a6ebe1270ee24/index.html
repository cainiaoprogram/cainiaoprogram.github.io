matlab查找最小值函数

一、查找最小值

function y=find_min(A)
a=min(A);

y=a;
end

或者自己写循环函数

function f=find_min2(A);
Len=length(A); min=A(1);
index=1;
for i=1:Len
    if min>A(i)
        min=A(i);
        index=i;
       
    end
end
 f=[min,index];
end
