C语言实战题目:使用for循环打印菱形详解

这是菱形的图案

for循环打印9层的菱形

打印方法-分为上三角形和下三角形来分别打印

1：for循环嵌套

#include<stdio.h>

#include<stdlib.h>

#include<Windows.h>

int main()

{

int i, j;

system("color f2");//f-亮白色背景；2-绿色前景

//打印上三角形

for (i = 1; i <= 5; i++)//i控制行数-上面为5行

{

for (j = i; j <= 4; j++)

//内循环1-控制打印的空格数（打印空格数时只需要考虑左边半部分的空格即可）

/*j从i的值开始，表示第几行打印5-i个空格数（如第一行打印4个空格j=1 j=2 j=3 j=4都符合j<=4的条件，打印四个空格）*/

//打印的空格数1-5行依次为4 3 2 1 0

{

printf(" ");

}

//打印*号-打印*号的个数需要计算：1-5行依次为1 3 5 7 9；刚好满足算式2*i-1

for (j = 1; j <= 2 * i - 1; j++)

{

printf("*");

}

printf("\n");

}

//打印下三角形

for (i = 1; i <= 4; i++)//i控制行数，下面为4行

{

for (j = 1; j <= i; j++)

/*由图形案例可知，下三角1-4行空格数依次为1 2 3 4空格数恰好与行数相等,判断条件为j<=i*/

{

printf(" ");

}

//打印*号个数，下三角的*号个数从1-4行依次为7 5 3 1，刚好为9减去行数的两倍（9-2*i）

for (j = 1; j <= (9 - 2 * i); j++)

{

printf("*");

}

printf("\n");

}

return 0;

}

#include<stdio.h>
#include<stdlib.h>
#include<Windows.h>
int main()
{
    int i, j;
    system("color f2");
    for (i = 1; i <= 5; i++)
    {
        for (j = i; j <= 4; j++)
        {
            printf(" ");
        }
        for (j = 1; j <= 2 * i - 1; j++)
        {
            printf("*");
        }
        printf("\n");
    }
    for (i = 1; i <= 4; i++)
    {
        for (j = 1; j <= i; j++)
        {
            printf(" ");
        }
        for (j = 1; j <= (9 - 2 * i); j++)
        {
            printf("*");
        }
        printf("\n");
    }
    return 0;
}

形式2：打印菱形

可自行决定打印的菱形的高度，以及打印的菱形由何种符号构成。

#include<stdio.h>
#include<stdlib.h>
#include<Windows.h>
int main()
{
    int i, j,high;
    char a;
    printf("请输入你想要菱形由什么构成\n");
    a = getchar();//熟悉getchar的使用
    printf("你想要菱形由几层构成\n");
    scanf("%d", &high);//手动输入菱形的层数（注意必须是奇数！）-可以打印多种菱形的个数
    system("cls");
    system("color f5");
    for (i = 1; i <= (high+1)/2; i++)
    {
        for (j = i; j <= (high-1)/2; j++)
        {
            printf(" ");
        }
        for (j = 1; j <= 2 * i - 1; j++)
        {
            printf("%c",a);
        }
        printf("\n");
    }
    for (i = 1; i<=(high-1)/2; i++)
    {
        for (j=1;j<=i;j++)
        {
            printf(" ");
        }
        for (j = 1; j <=(high-2*i); j++)
        {
            printf("%c",a);
        }
        printf("\n");
    }
    return 0;
}
