    while start<=end:
        mid=(start+end+1)>>1
        if num>arr[end] or num<arr[0]:

            break
        elif arr[mid]==num:
            print(mid)
            break
        elif arr[mid]>num:
            return find(num,arr,start,mid-1)
        else:
            return find(num,arr,mid,end)
