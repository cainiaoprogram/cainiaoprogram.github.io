#include<math.h>
main()
{

	printf("请输入你的分数：");
	scanf("%d",&score);
	printf("你的成绩为%c\n",score>=90&&score<=100?'A':(score>=70?'B':(score>=60)?'C':'D'));
		
	return 0;

}
