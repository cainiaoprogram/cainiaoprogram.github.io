php使用sqlite(超简单版)

一、直接上代码

查询：

public function select()
    {
        try {
            $maindir = 'D:\xxx.db';
            $main = new \SQLite3($maindir);
            if (!$main) {
                echo 'code：' . $main->lastErrorCode();
                echo 'Error：' . $main->lastErrorMsg();
                $main->close();
                die;
            }
            $sql = 'select * from 表名';
            $result = $main->query($sql);
            while ($row = $result->fetchArray(SQLITE3_ASSOC)) {
                $arr = $row['字段名'];
                var_dump($arr);
            }
            $main->close();
        } catch (\Exception $e) {
            if (isset($main)) {
                $main->close();
            }
            var_dump($e->getMessage());
        }
    }
