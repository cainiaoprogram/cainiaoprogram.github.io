java获取当前日期的前一天,年，月等

Calendar ca = Calendar.getInstance();//得到一个Calendar的实例 
ca.setTime(new Date()); //设置时间为当前时间 
ca.add(Calendar.YEAR, -1); //年份减1 
Date lastMonth = ca.getTime(); //结果
//求前一月ca.add(Calendar.MONTH, -1)，
//前一天ca.add(Calendar.DATE, -1)
