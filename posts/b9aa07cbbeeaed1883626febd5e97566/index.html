写二进制显示的文本

    QString binData = QDateTime::currentDateTime().toString("yyyy-MM-dd");
    char* cTemp=array.data();
    uint8_t* ucTemp =reinterpret_cast<uint8_t*>(cTemp);

    //先将接收到的原始数据写文件
    for(int i=0;i<array.size();i++)
    {
        binData+=" "+QString("%1").arg(QString::number(ucTemp[i],2),8,'0');
    }

    binData += "\n\n\n\n\n";

    //写测试文件
    QString strNameTmp = "D:/Test.log";

    QFile fileTmp;
    fileTmp.setFileName(strNameTmp);

    if(fileTmp.open(QIODevice::WriteOnly | QIODevice::Append | QIODevice::Text))
    {
           fileTmp.write(binData.toUtf8());
           fileTmp.close();
    }
