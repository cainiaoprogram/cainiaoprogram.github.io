	for (i = 0; i < len; i++)
	{
		if (0 == strcmp( Arr[i].Name, Name))

			num = i;
		}

6.按成绩打印学生排名

void ScoreTotal(struct Student*  Arr, int len)
{
	struct Student temp;
	int max;
	for (int i = 0; i < len - 1; i++)
	{
		max = i;
		for (int j = i + 1; j < len; j++)
		{
			if ( Arr[j].Total >  Arr[max].Total)
			{
				max = j;
			}
		}
		if (max != i)
		{
			temp =  Arr[max];
			 Arr[max] =  Arr[i];
			 Arr[i] = temp;
		}
	}
	Output( Arr, len);
}

使用冒泡排序，来实现排名！
