彩色图像高频与低频成分的分解

import torchgeometry
import cv2
import torch
import numpy as np
def calc_Freq(torch_img, kernel=3):
    if kernel == 3:
        sigma = 3
    elif kernel == 5:
        sigma = 1.5
    else:
        sigma = 1
    lowFreq = torchgeometry.image.gaussian_blur(
        torch_img, (kernel, kernel), (sigma, sigma)
    )
    highFreq = torch_img - lowFreq
    return lowFreq, highFreq
