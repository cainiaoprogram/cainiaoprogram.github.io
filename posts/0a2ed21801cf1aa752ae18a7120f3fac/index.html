np.concatenate()函数类似于torch.cat()函数

import random
import numpy as np

random.seed(0)
a = np.random.randn(2,3)
b = np.random.randn(2,3)
c = np.concatenate((a, b), axis=0)
d = np.concatenate((a, b), axis=1)

print("a:\n-----\n", a)
print("b:\n-----\n", b)
print("c:\n-----\n", c)
print("d:\n-----\n", d)
