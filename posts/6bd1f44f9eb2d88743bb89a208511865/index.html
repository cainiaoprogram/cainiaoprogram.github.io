js金额转换万元显示

    methods: {

      formatNumber(num) {
        num = Number(num);
        if (num == 0) {
          return num + "";
        } else {
          if ((num / 10000).toFixed(2) == 0) {
            //小于100的保留2位
            return (num / 100).toFixed(4)
          } else {
            // parseFloat() 去掉后面不用的0，如50.00
            //大于100的保留2位
            return parseFloat((num / 10000).toFixed(2)) + '万'
          }
        }
      },

}
