1.最长公共子序列

问题描述

给定序列X={x1,x2,…xm}和Y={y1,y2,y3…yn},找出X和Y的最长公共子序列

最长公共子序列的结构

假设序列X={x1,x2,… , xm }和Y={y1, y2 , … , yn}的一个最长公共子序列为
Z= {z1,z2 ,… ,zk }，则有
假设序列X={x1,x2,… , xm}和Y={y1, y2 , … , yn}的一个最长公共子序列为
Z={z,z2 ,.….,zk }
1)若xm=yn.则zk=xm =yn,且Zk-1是Xm-1，和Yn-1的最长公共子序列。
2)若xm≠ yn,且zk ≠ xm，则Z是Xm-1,和Yn的最长公共子序列.
3)若xm≠yn且zx ≠yn，则Z是X和Yn-1的最长公共子序列.
其中, Zk-1={z1, z2 ，… ,zk-1}，Xm-1={x1,x2,…xm-1},Yn-1={y1, y2 … yn-1}

子问题的递归函数

设Xi= {x1, x2 ,… ,xi } , Yj = {y1, y2 ,… , yi},用c[i]lj]记录Xi和Yj的最长公共子序列的长度，则原问题的最优值为c[m][n]。

#include<iostream>
#include<stdio.h>
using namespace std;

int LCSLength( char* x, char* y, int** c,int** b, int m, int n){
	//循环求解子结构c[i][j]，决策数组b[i][j] 
	int i,j;
	for(i=1;i<=m;i++)c[i][0]=0;//j=0 
	for(j=0;j<=n;j++)c[0][j]=0; //i=0
	for(i=1;i<=m;i++)
	   for(j=1;j<=n;j++)
	       if(x[i]==y[j])
	           {c[i][j]=c[i-1][j-1]+1; b[i][j]=0;}
            else if(c[i-1][j]>=c[i][j-1])
	          {c[i][j]=c[i-1][j]; b[i][j]=-1;}
			else {c[i][j]=c[i][j-1]; b[i][j]=1;}
	return c[m][n];
	}

void BestSolution(int** b,char* x, int i, int j){
	//最优解 
	if(i==0||j==0) return; 
	if(b[i][j]==0)
	{
	BestSolution(b,x,i-1,j-1);cout<<x[i];
	}
	else if(b[i][j]==-1)BestSolution(b,x,i-1,j);
	else BestSolution(b,x,i,j-1);
}


int main(){
	char*s1,*s2;
	int**m,**d;
	int i,l1,l2;
	cout<<"请分别输入两个序列的长度："<<endl;
	cin>>l1>>l2;
	s1=new char[l1+1];s2=new char[l2+1];
	m=new int *[ l1+1];d=new int *[l1+1];
	for(i=0;i<=l1;i++)
	{
	m[i]=new int [l2+1];d[i]=new int [l2+1];
	}
	cout<<"请分别输入两个序列："<<endl;
	for(i=1;i<=l1;i++)cin>>s1[i];
	for(i=1;i<=l2;i++)cin>>s2[i];
	cout<<"Best Value: "<< LCSLength(s1,s2, m,d,l1,l2)<<endl;
	cout<<"Best Solution: "; BestSolution(d,s1,l1,l2); cout<<endl;

	return 0;
}

#include<iostream>
#include<stdio.h>
using namespace std;

int MaxSum ( int* x, int n, int &st, int &ed){
	int *b,*c,mv;
	int i;
	b=new int[n+1]; c=new int [n+1];
	//循环求解b[j] c[j] 
	b[1]=x[1]; c[1]=0;
	for( i=2;i<=n;i++)
	if(b[i-1]>0){b[i]=b[i-1]+x[i];c[i]=1;}
	 else {b[i]=x[i];c[i]=0; }
	 
	 //找出b[j]的最大值 
	ed=1; mv=b[1];
	for(i=2;i<=n;i++)
	  if(b[i]>mv){mv=b[i];ed=i; }
	  
	//找出最大字段和的开始位置 
	st=ed;
	while(c[st])st--;
	return mv ;

}

3.最长单调递增子序列

问题描述

找出由n个数组成的序列的最长单调递增子序列。(时间复杂性为O(n^2))

最长单调递增子序列的结构

解:令n个数为x={x1，x2 , … ,xn}，设mj是以xj开头的一个最长单调递增子序列，其长度为cj。
则原问题求的最优值是:max {cj}(l≤j≤n)．显然,cj有下面递归式成立:

为构造最佳解,令用bj记住mj中xj的后一个元素的下标，若后面无元素记为0。
时间复杂性:O(n^2)

#include<iostream> 
#include<stdio.h>
using namespace std;
int Increase(int x[],int n,int *bx){
int *b,*c;
b=new int [n+1];//存储后续元素的下标 
c=new int [n+1];//最优值数组 
c[n]=1;b[n]=0;//初始化 
for(int j=n-1;j>=1;j--){
	c[j]=1;b[j]=0;
   for(int k=j+1;k<=n;k++)
      if(x[j]<=x[k]&&(c[k]+1)>c[j])c[j]=c[k]+1,b[j]=k;
}
//找c最大数及对应位置
int ml=c[1],st=1;
for(int j=2;j<=n;j++) {
	if(c[j]>ml)ml=c[j],st=j;
}
  //以st开头的最长单调递增子序列
  int j=0;
  while(st){
  	bx[++j]=x[st];
  	st=b[st];
  	}
  	return ml; 
}

int main(){
	int *x,*bx,n;
	cout<<"请输入数组的长度n："<<endl;
	cin>>n;
	x=new int[n+1];bx=new int[n+1]; 
	cout<<"请输入数组x："<<endl;
	for(int j=1;j<=n;j++)cin>>x[j];	
	int k=Increase(x,n,bx);
	cout<<"最长单调递增子序列为："<<endl;
	cout<<k<<endl;	
	for(int j=1;j<=k;j++)cout<<bx[j]<<'\t';
	cout<<endl;
	return 0;
}
