postgresql 查询所有表名

SELECT   tablename   FROM   pg_tables 
WHERE   tablename   NOT   LIKE   'pg%'
AND tablename NOT LIKE 'sql_%' 
 ORDER   BY   tablename;

 

或

select tablename from pg_tables where schemaname='public'
