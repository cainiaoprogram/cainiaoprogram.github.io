C语言实现栅栏式加密算法

 根据栅栏式加密的一般方法，写下如下代码，算是暴力解决加密及解密的算法（无指针，设有循环可持续加密/解密）：

#include <stdio.h>
#include <string.h>
#define N 60
void test();
void key();
int main (){
	int t=1;
	while(t){
		printf("请选择加密/解密/退出（1/2/0）");
		scanf("%d",&t);
		getchar();
		if(t==1){
			test();
		}
		else if(t==2){
			key();
		}
		else if(t==0){
		}
		else{
			printf("输入有误，请重新选择");
		}
		}
	return 0;
}

void test(){
	int i,k,j,d,p,z;
	char word[N],a[5][30];
	printf("请输入明文：");
	gets(word);
	p=strlen(word);
	printf("请选择加密深度：");
	scanf("%d",&d);
	getchar();
	for(i=0;i<p;i++){//写入二维数组
		k=i%d;//k a[k][]
		j=i/d;//i a[][i]
		a[k][j]=word[i];

	}
	k=0;
	z=p%d;
	if(z==0){
		z=-1;}
	for(i=0;i<z;i++){
		for(j=0;j<=(p/d);j++){
			word[k]=a[i][j];
			k++;
		}
	}
	if(z<d){
		for(;i<d;i++){
			for(j=0;j<(p/d);j++){
				word[k]=a[i][j];
				k++;
		}
		}
	}
	printf("栅栏式密文：%s",word);
	printf("\n");
}
void key(){
	int i,k,j,d,p,z,g=0;
	char word[N],b[5][30];
	printf("请输入密文：");
	gets(word);
	p=strlen(word);
	printf("请选择加密深度：");
	scanf("%d",&d);
	getchar();
	p=strlen(word);//p个字符
	k=p/d;//k列是正好的
	z=p%d;//z行是正好的
	if(z==0){//二维数组正好满
		for(i=0;i<d;i++){//拆解密文入二维数组
			for(j=0;j<k;j++){
				b[i][j]=word[g];
				g++;
			}
		}
		g=0;
		for(i=0;i<k;i++){//解密好的明文写入word
			for(j=0;j<d;j++){
				word[g]=b[j][i];
				g++;
			}
		}
	}
	else{//二维数组不满
		for(i=0;i<z;i++){//可将二维数组横列写满的写入二维数组
			for(j=0;j<=k;j++){
				b[i][j]=word[g];
				g++;
			}
		}
		for(i=z;i<d;i++){//不可将二维数组横列写满的写入二维数组
			for(j=0;j<k;j++){
				b[i][j]=word[g];
				g++;
			}
		}
		g=0;
		for(i=0;i<k;i++){//二维数组竖列写满的写入word明文
			for(j=0;j<d;j++){
				word[g]=b[j][i];
				g++;
			}
		}
		for(j=0;j<z;j++){//二维数组竖列不满的写入word明文
			word[g]=b[j][k];
			g++;
		}
	}
	printf("栅栏式明文：%s",word);
	printf("\n");
}
