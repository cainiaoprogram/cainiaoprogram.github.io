jsp内置对象的详解

<%@ page contentType="text/html;charset=UTF-8" language="java" %>
<html>
<head>
    <title></title>
</head>
<body>
<%--内置对象 --%>
<%
    pageContext.setAttribute("name1","lmh1");//保存的数据再页面中有效

    request.setAttribute("name2","lmh2");//保存的数据在一次请求中有效，请求转发会携带这个数据
    //客户端向服务器发送请求，产生的数据用户看完后就没用了 比如：新闻
    session.setAttribute("name3","lmh3");//保存的数据在一次会话中有效，打开到关闭浏览器为一次会话
    //客户端向服务器发送请求，产生的数据用完还可以用 比如：购物车
    application.setAttribute("name4","lmh4");//保存的数据在服务器中有效，打开到关闭服务器
    //客户端向服务器发送请求，产生的数据，用户用完 其他用户也可以用 比如聊天数据
%>
<%--通过pageContext取出保存的值 --%>
<%--脚本片段的代码会被原封不动生成到.JSP.java中   要求里面的代码必须保证Java代码语法的正确性 --%>
<%
    //通过寻找的方式来
    //从底层到高层（作用域）：page->request->session->application
    //JVM 双亲委派机制
    String name1 = (String) pageContext.findAttribute("name1");
    String name2 = (String) pageContext.findAttribute("name2");
    String name3 = (String) pageContext.findAttribute("name3");
    String name4 = (String) pageContext.findAttribute("name4");
    String name5 = (String) pageContext.findAttribute("name5");//没有存在
%>
<%--通过EL方式输出 ${}}--%>
<h1>取出的值</h1>
<h2>${name1}</h2>
<h2>${name2}</h2>
<h2>${name3}</h2>
<h2>${name4}</h2>
<h2><%=name5%></h2>
</body>
</html>
