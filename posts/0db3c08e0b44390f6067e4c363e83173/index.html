def hanoi(n, x, y, z):
    if n == 1:
        print("这里是if开始")
        print(f'n={n}啊 ' + x, '-->', z)  # 如果只有 1 层，直接将金片从 x 移动到 z
        print("这里是if结束")
    else:
        print("这里是else开始")
        hanoi(n - 1, x, z, y)  # 将 x 上的 n-1 个金片移动到 y
        print(f'n={n}嗯 ' + x, '-->', z)  # 将最底下的金片从 x 移动到 z
        hanoi(n - 1, y, x, z)  # 将 y 上的 n-1 个金片移动到 z
        print("这里是else结束")
