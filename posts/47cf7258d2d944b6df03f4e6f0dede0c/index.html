service文件中添加显卡判断

需求：要求判断机器显卡正常则启动服务，反之则不启动服务。
在service中添加下面这行即可

ExecStartPre=/usr/bin/nvidia-smi

源文件：

[Unit]
Description=Lotus Worker C2
After=network-online.target
Requires=network-online.target

[Service]
Environment=WORKER_DIR=/root/.lotuswork/C2
Environment=SEALER_API_INFO=xxxxxx
Environment=IPFS_GATEWAY="https://proof-parameters.s3.cn-south-1.jdcloud-oss.com/ipfs/"
Environment=BELLMAN_MULTI_GPU=1
Environment=CUDA_VISIBLE_DEVICES=0
Environment=RUST_LOG=Info
ExecStartPre=/usr/bin/nvidia-smi
ExecStart=/root/worker/lotus-worker  --workerrepo=/root/.lotuswork/C2 run --listen=0.0.0.0:5901 --precommit1=false --precommit2=false --commit=true --addpiece=false --no-check-param=true 
Restart=always
RestartSec=30
StandardOutput=journal
StandardError=inherit

[Install]
WantedBy=multi-user.target
