今天学了一个猜数游戏的小程序，其中就用到了rand（）生成随机数的作用。下面是程序的具体情况：

#include<stdio.h>
#include<math.h>
int main()
{
    int a,b,c,d;
    printf("boys,please enter your lovely number a,c\n");
    scanf("%d%d",&a,&b);
    c=sqrt(a*b+a)+a-b;
    d=rand()%100;
    if(c>d)
    {
        printf("hi,boys");
        printf("you guess the number c is not right");
    }
    else if(c<d)
    {
        printf("hi,boys");
        printf("you guess the number c is not right");
    }
    else
    {
        printf("hi,boys");
        printf("you guess the number c is right");
    }
    return 0;
}
