import jieba
import wordcloud
import matplotlib.pyplot as plt

file = open('二十大报告全文.txt', encoding='utf-8')
str = file.read()
file.close()

removes = ['和','的']

str_list = jieba.lcut(str)
text = ' '.join(str_list)
cloud = wordcloud.WordCloud(font_path='simhei', stopwords=removes, background_color='pink').generate(text)

plt.imshow(cloud)
plt.axis('off')
plt.savefig('复习词云.jpg')
