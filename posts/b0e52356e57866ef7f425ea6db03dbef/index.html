矩阵转置——C++实现

C++代码：

#include<iostream>
using namespace std;

int main() {
	int m, n, array[50][50], result[50][50];
	cin >> m >> n;

	for (int i = 0; i<m; i++) {
		for (int j = 0; j<n; j++) {
			cin >> array[i][j];
			result[j][i] = array[i][j];
		}
	}
	cout<<endl;

	for (int i = 0; i<n; i++) {
		for (int j = 0; j<m; j++) {
			cout << result[i][j] << " ";
		}
		cout << endl;
	}
}
