IOS webView播放视频时进入页面自动播放的问题

主要解决用urlStr通过 webView 去播放视频时进入界面就自动播放的问题
 NSString *htmlStr = [NSString stringWithFormat:
      @"<!DOCTYPE HTML><html><body> 
              <video src=\"%@\" controls=\"controls\"> 
                     该浏览器不支持
              </video> </body> </html>",urlStr];
 [self.videoView loadHTMLString:htmlStr baseURL:nil];
