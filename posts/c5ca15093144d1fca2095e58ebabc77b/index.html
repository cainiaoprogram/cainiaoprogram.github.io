3.4.1.3、for 循环方式和调用math函数方式计算百分比形成综合程序如下所示

#define _CRT_SECURE_NO_WARNINGS

#include <stdio.h>
#include <stdlib.h>
#include <math.h>

int main()
{
	system("color 3E");

	double p, r;
	int n;

	p = 1.0;
	r = 0.09;
	n = 10;

	// for 循环方式
	for (int i = 0; i < n; i++)
	{
		p = p * (1 + r);
	}

	printf("%.3lf\n", p);
	
	// 调用math函数方式
	p = 1.0;
	r = 0.09;
	n = 10;

	p = pow(1 + r, n);
	printf("%.3lf\n", p);

	system("pause");
	return 0;
}
