VS简易示波器（四）：绘制波形

示波器的基本框架搭建好了，接下来开始绘制波形（调试上位机软件阶段，我以产生随机数为例，以下只为部分关键代码）：

1） 在子属性对话框中添加8个通道选择复选框；

2） 定义复选框控件变量并在OnInitDialog()中关联：

CButtonm_checkDn[8];

for (int i = 0; i <8; i++){

        m_checkDn[i].SubclassDlgItem(IDC_CHECK_D0+ i, this);        //关联控件

}

3） 在发送按钮响应函数中将通道设置参数发送到主对话框：

USHORT usData = 0;

UCHAR temp = 0x01;

for (int i = 0; i <8; i++){

        if (m_checkDn[i].GetCheck()){

               ucCmd = ucCmd | temp;

        }

        temp = temp << 1;

}

m_cmddata.m_ucCmd = 0xFF;

m_cmddata.m_usData = usData;

::SendMessage(this->GetParent()->m_hWnd,WM_FDM_MESSAGE, DWORD(&m_cmddata), DWORD(0));

4） 在主对话框中接收信息函数中，根据参数启动定时器：

cmddata* m_cmddata = (cmddata*)wParam;

m_ucCmd = m_cmddata->m_ucCmd;

m_usData = m_cmddata->m_usData;

if (m_ucCmd == 0xFF)

SetTimer(1, m_iTimer, NULL);

5） 添加绘制波形函数DrawGraph(CDC *pDC, SHORT*sData, int iLen)：

if (usData == 0)

        return;

 

CPen *pPen = new CPen();                                           //创建画笔对象

pPen->CreatePen(PS_SOLID, 1, crPen[0]);

CGdiObject *pOldPen =pDC->SelectObject(pPen);

 

int i, j;

USHORT usData= m_usData;

for (j = 0; j < 8; j++){

        if (usData & 0x0001){                              //通道有效且勾选则绘制曲线

               pPen->DeleteObject();

               pPen->CreatePen(PS_SOLID, 1,crPen[j]);

               pDC->SelectObject(pPen);

 

               pDC->MoveTo(XX0,YY0);               //省略。。。

               for (i = 1; i < iLen; i++){

                      pDC->LineTo(XXi,YYi);           //省略。。。

               }

        }

usData = usData>>1;

}

6） 在主对话框.h函数中添加变量代码：

SHORT*uDrawBuf;     //数据缓存指针

int m_iShowLen= 1024;      //绘图长度

7） 在OnPaint()函数中添加代码：

DrawGraph(&dc,uDrawBuf, m_iShowLen);

8） 添加重新绘制函数ReDrawGraph()：

CRect rect;

rect.left = WINDOW_OFFSET;

rect.top = WINDOW_OFFSET;

rect.right = rect.left + WINDOW_WIDTH;

rect.bottom = rect.top + WINDOW_HIGHT;

InvalidateRect(rect, true);   //擦除设置背景区域

9） 添加定时器响应函数OnTimer及代码：

UCHAR* inData = new UCHAR[RECVLEN];

ZeroMemory(inData, RECVLEN);

 

srand((unsigned)time(NULL));

for (int i = 0; i < RECVLEN; i++){

        inData[i] = rand() % 256;

}

delete[] inData;

ReDrawGraph();
