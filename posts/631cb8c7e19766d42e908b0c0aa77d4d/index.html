imread改变图片显示尺寸

挺简单的，但是不会，还查了好久，伤心，写给自己查看

int main()
{
    Mat girl = imread("F:/VSMaven/LocateQr/Location/Location/phone.jpg", IMREAD_REDUCED_COLOR_2); //载入图像到Mat
    if (girl.empty()) {
        printf("没有此图片，请检查路径是否正确");
        return -1;
    }
    else {
        namedWindow("enhanced", 0);
        resizeWindow("enhanced", 300, 400);//显示的就是300*400的大小，可以拖拽
        imshow("enhanced", girl);//显示名为 "enhanced"的窗口  
    }
    waitKey(0);
    return 0;

是else中的那三条语句，尺寸的大小可以自己设置
