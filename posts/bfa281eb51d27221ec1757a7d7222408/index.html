jquery实现60秒倒计时

代码：

 var time = 60;
//倒计时
function getRandomCode() {
   if (time === 0) { 
       time = 60;
       return;
   } else { 
       time--;
       $('#time i').text(time);
   } 
   setTimeout(function() { 
       getRandomCode();
   },1000);
}
