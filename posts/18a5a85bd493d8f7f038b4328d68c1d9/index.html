list列表，tuple元组，dictionary字典

 

# list列表
list1 = [1, 2, 3, 4, 5, 6]

# pop()代表删除列表最后一个元素，pop(i)代表删除索引i的元素
# list1.pop()
# insert()代表在列表索引i处添加j
# list1.insert(5, 7)
# append()代表列表末尾添加元素
# list1.append(7)
# sort()排序
# reverse()反转
# remove(i)代表代表删除列表中的i
# list1.remove(6)
# print(list1)

# tuple元组(不能删除元素)
tuple1 = (1, 2, 3)
list1 = list(tuple1)
list2 = [1, 2, 3]
tuple2 = tuple(list2)
print(tuple2)
print(list2)

# dictionary
dict1 = {'name': "Link", 'height': 173, 'weight': 140}
dict1["gender"] = "male"
print(dict1.keys())
print(dict1.values())
print(dict1)
dict1.pop("gender")
print(dict1)
