  private void textBox1_KeyDown(object sender, System.Windows.Forms.KeyEventArgs e)
     {
        //if条件检测按下的是不是Enter键
           if(e.KeyCode==Keys.Enter)
           {
                 MessageBox.Show("您在textBox1里按下了回车键");
           }
     }
