jquery ajax success beforeSend

jQuery.ajax({       //Ajax函数
          type: 'get',   //定义http方法
          data: 'version_id='+version_id ,
          url: "/parent_package_versions/show_baseline_details",   //请求的URL
          beforeSend: function (){
            jQuery('#ajax-indicator span').text('处理中，请稍等...');
            Element.show('ajax-indicator');
          },
          success: function() {
            Element.hide('ajax-indicator');
            tb_show("<%=l(:label_parent_package_version_details)%>",'##TB_inline?height=500&width=600&inlineId=parent_package_version_details' , "thickbox");
          }

        });


jQuery ajax异步处理时，发送异步请求前，执行beforeSend，异步操作成功后，执行success
