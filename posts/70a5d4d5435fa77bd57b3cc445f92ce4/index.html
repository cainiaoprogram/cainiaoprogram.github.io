shell 猜商品价格

#!/bin/bash
aaa=$(expr $RANDOM % 1000)
bbb=0
echo "商品价格为0-999，猜实际价格"
while true
do
   read -p "请输入你猜的及价格： "  n
   ((bbb++))
if [ $aaa -eq $n ]
    then
    echo "恭喜你答对了！！！"
    echo "你一共答了$bbb 次"
    exit 0

elif [ $n -gt $aaa ]
    then
    echo "太高了"
    else
    echo "太低了"
fi
done
