遍历数组查找数组有没有需要的数值

public class demo8 {
    /*
    查找
    需求：已知一个数组arr={19,28,37,46,50};键盘录入一个数据，查找该数据在数组中的索引，并在控制台输出
    分析：键盘录入一个数据后，将该数据与数组中的每一个值进行比较，如果数据相等，返回该数据值对应的索引即可。
     */
    public static void main(String[] args) {
        int[]arr={1,2,3,5,6};
        System.out.println("请输入您要查找的元素：");
        Scanner sc=new Scanner(System.in);
        int x=sc.nextInt();
        int index=-1;//假设要查找的值不存在
        for (int i = 0; i <arr.length ; i++) {
            if(x==arr[i]){
              index=i;
              break;
            }

        }
        System.out.println("在该数组中索引值为："+index);
    }
}
