执行过的 SQL 语句

SELECT *
  FROM V$SQLAREA T
 WHERE ROWNUM <= 50
   and t.PARSING_SCHEMA_NAME = 'ZTENET'
   and t.SQL_TEXT like '%ZTE_FMT_SEAL%'
 ORDER BY T.LAST_ACTIVE_TIME DESC;

执行中的 SQL 语句

SELECT B.SID      ORACLEID,
       B.USERNAME 登录ORACLE用户名,
       B.SERIAL#,
       SPID       操作系统ID,
       PADDR,
       SQL_TEXT   正在执行的SQL,
       B.MACHINE  计算机名
  FROM V$PROCESS A, V$SESSION B, V$SQLAREA C
 WHERE A.ADDR = B.PADDR
   AND B.SQL_HASH_VALUE = C.HASH_VALUE
   and b.USERNAME = 'ZTENET';
