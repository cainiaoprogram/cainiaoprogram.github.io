  1. Сй┐уће has()Тќ╣Т│Ћ№╝їтЈ»С╗ЦТЪЦУ»бТЪљС║ЏТЮАС╗ХСИІуџётЁ│УЂћТЪЦУ»бТЋ░ТЇ«№╝Џ
//УјитЈќтГўтюетЁ│УЂћС╣ду▒ЇуџёућеТѕитѕЌУАе(УеђСИІС╣ІТёЈ№╝џУЄ│т░ЉСИђТюгС╣д)
$users = User::has('book')->get();
return $users;
//УјитЈќтГўтюетЁ│УЂћС╣ду▒Ї(т╣ХУХЁУ┐Є 3 ТЮА)уџёућеТѕитѕЌУАе

  1. Сй┐уће whereHas()Тќ╣Т│Ћ№╝їтѕЏт╗║жЌГтїЁТЪЦУ»б№╝Џ
//whereHas жЌГтїЁућеТ│Ћ

//У┐ЎжЄї$query Тў» book УАе№╝їжђџУ┐Є user_id ТЪЦУ»б№╝їУ┐ћтЏъ user УАеТЋ░ТЇ«
$query->where('user_id', 19);
})->get();
return $users;
  1. Сй┐уће doesntHave()Тќ╣Т│Ћ№╝їтЇ│ has()уџётЈЇтљЉТЊЇСйю№╝Џ
//УјитЈќСИЇтГўтюетЁ│УЂћС╣ду▒ЇуџёућеТѕитѕЌУАе№╝їжЌГтїЁућеТ│Ћ№╝џwhereDoesntHave()
$users = User::doesntHave('book')->get();
return $users;
  1. Сй┐уће withCount()Тќ╣Т│Ћ№╝їтЈ»С╗ЦУ┐ЏУАїтЁ│УЂћу╗ЪУ«А№╝Џ
//тЁ│УЂћу╗ЪУ«А№╝їС╝џУЄфтіеу╗ЎСИђСИф book_count тГЌТ«х
//у╗ЪУ«АТ»ЈСИфућеТѕиТюЅтцџт░ЉТюгС╣д
$users = User::withCount('book')->get();
return $users;
//у╗ЎтцџСИфтЁ│у│╗Ти╗тіау╗ЪУ«А№╝џprofile_count№╝їbook_count
$users = User::withCount(['profile', 'book'])->get();
return $users;
//тЁ│УЂћу╗ЪУ«АтєЇу╗ЊтљѕжЌГтїЁУ┐ЏУАїуГЏжђЅ№╝їУ┐ўтЈ»С╗ЦУ«Йуй«тѕФтљЇ
$users = User::withCount(['profile', 'book' => function ($query) {
//У┐ЎжЄїжЎљтѕХУбФу╗ЪУ«АуџёУ«░тйЋ
$query->where('user_id', 19);
}])->get();
return $users;
