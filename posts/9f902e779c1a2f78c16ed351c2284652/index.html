删除重复字符 C++

/*
Description  
         给定一个字符串，将字符串中所有和前面重复多余的字符删除，其余字符保留，输出处理后的字符串。需要保证字符出现的先后顺序。
Prototype
         int GetResult(const char *input, char *output)
Input Param 
         input     输入的字符串
Output Param 
         output    输出的字符串
Return Value
         0         成功
         -1        失败及异常
*/
int GetResult(const char *input, char *output)
{
	if(input==NULL||output==NULL){
		return -1;
	}
	char exist[1024]={0};
	int length=strlen(input);
	int out=0;
	for(int i=0;i<length;i++){
		if(!contains(exist,input[i])){
			output[out]=input[i];
			exist[out]=input[i];
			out++;
		}
	}
	output[out]='\0';
    return 0;
}
/*
验证input中是否含有c
有 返回1
无 返回0
*/
int contains(const char *input,char c){
	int length=strlen(input);
	if(!input) {
		return 0;
	}
	for(int i=0;i<length;i++){
		if(input[i]==c){
			return 1;
		}
	}
	return 0;

}
