用Swap函数交换同一个数组的两个元素的值

用Swap函数交换同一个数组的两个值（C语言）

#include<stdio.h>
void Swap(int a[], int i, int j) {
	int temp = a[i];
	a[i] = a[j];
	a[j] = temp;
}
void main() {
	int a[] = { 1,2,3,4,5,6,7,8,9 };
	Swap(a, 2, 3);
	printf("%d\n", a[2]);
	printf("%d\n", a[3]);
}
