C++保存数据到本地文件

C++ 保存数据到本地文件常用的有两种方法：

1， 使用ofstream

#include <fstream>

int main()
{
    std::ofstream outFile;
    //打开文件
    outFile.open("F:\\Test.txt");
    int arr[5] = {1, 2, 3, 4, 5};
    int i;
    for(i = 0; i < 5; i++)
    {
        //写入数据
        outFile << arr[i];
    }
    //关闭文件
    outFile.close();
}

2，使用fprintf

#include <stdio.h>

int main()
{
    //打开文件
    FILE *outFile = fopen("F:\\Test.txt", "w");
    int arr[5] = {1, 2, 3, 4, 5};
    int i;
    for(i = 0; i < 5; i++)
    {
        //写入数据
        fprintf(outFile, "%d, ", arr[i]);
    }
    //关闭文件
    fclose(outFile);
}

也可以按16进制输出：

fprintf(outFile, "%02x, ", arr[i]); //02 表示不足两位，前面补0输出；如果超过两位，则实际输出
