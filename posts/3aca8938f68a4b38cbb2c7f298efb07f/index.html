numpy学习之—— np数据存储和加载（内存映射）


import numpy as np

a = np.array([1,2,3])
b = np.array([1,2,3,4])

np.savez("save.npz", a, b)

print("save ok ....")

import numpy as np

r = np.load("save.npz")

a = r['arr_0']
print (a)

b = r['arr_1']
print(b)
