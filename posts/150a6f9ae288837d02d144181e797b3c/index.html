#include<bits/stdc++.h>
#define ll long long
using namespace std;
ll f(ll j){
    ll ans=1;
    for(int i=1;i<=j;i++){
        ans*=i;
    }
    return ans;
}
int main()
{
    ll p;
    cin>>p;
    ll ans=0;
    ll i=10;
    while(p!=0){
        if(f(i)>p)i--;
        else if(f(i)<=p){
            p-=f(i);
            ans++;
        }
    }
    cout<<ans<<endl;
}

#include<bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
    ll n,k;
    cin>>n>>k;
    ll k2=k%n;
    ll num0=k/n;
    ll num[n];
    ll ans[n];
    for(ll i=0;i<n;i++){
        cin>>num[i];
        ans[i]=num[i];
    }
    sort(num,num+n);
    int add=0;
    ll key=num[k2-1];
    if(k2>0)add++;
    for(ll i=0;i<n;i++){
        if(ans[i]<=key)cout<<num0+add<<endl;
        else cout<<num0<<endl;
    }
return 0;
}
