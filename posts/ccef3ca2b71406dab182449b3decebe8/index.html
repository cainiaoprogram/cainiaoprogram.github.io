设计一个函数：将一维整型数组中的元素逆序存放

void reverseArray(int a[],int n)
{
    for(int i = 0; i < (n/2); i++)
   {
       int temp = a[i];
       a[i] = a[n-1-i];
       a[n-1-i] = temp;    
   }
}


int main()
{
    printf("请输入5个整数\n");
    int array[5];
    for(int i=0;i<5;i++)
    {
        scanf("%d",&array[i]);
        if(getchar()=='\n')break;
    }
    
    reverseArray(array,5);
    
    for(int i = 0;i < 5; i++)
    {
        printf("%d\t", array[i]);
    }
    return 0;
}
