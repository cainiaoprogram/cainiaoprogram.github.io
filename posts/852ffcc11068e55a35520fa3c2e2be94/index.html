opencv同时显示两张图片

import cv2

import numpy as np

def img_show(name, img):
    cv2.imshow(name, img)
    cv2.waitKey(0)
    

img1=cv2.imread(path1)
img2=cv2.imread(path2)

img1 = cv2.resize(img1, (640, 640))
img2 = cv2.resize(img2, (640, 640))

new_img = np.hstack([img1, img2])

img_show('new_img', new_img)
