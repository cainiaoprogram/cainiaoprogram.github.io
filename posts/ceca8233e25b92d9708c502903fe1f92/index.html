编写一个求n个数的平均值及最小值的程序。要求：程序执行时，用户先由键盘输入n的值，之后输入n个数，最后，程序显示平均值和最小值。

import java.util.*;  
class Nnumber {
    public static void main (String args[ ]){
    	double sum=0;
    	int n=0,b=0;
        System.out.println("请输入n的值：");
        Scanner reader = new Scanner(System.in);
		 n = reader.nextInt();
        System.out.println("请输入n个数，每个数字用回车结束");
        int a= reader.nextInt();
        sum=a;
        for(int i=1;i<n;i++){
        	b= reader.nextInt();
          	sum = sum+b;        	
        		if(a>b){
        			a=b;}
       	}
        System.out.println("一共"+n+"个数，最小值是："+a);  
        System.out.println("平均值是："+sum/n);                        
    }
}
