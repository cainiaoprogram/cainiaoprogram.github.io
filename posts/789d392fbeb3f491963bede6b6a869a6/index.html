字符串/字节互转

#bytes object 

byte = b"byte example" 

# str object 

str = "str example" 

# str to bytes 字符串转字节 

bytes(str, encoding="utf8") 

# bytes to str 字节转字符串 

str(bytes, encoding="utf-8") 

# an alternative method 

# str to bytes 字符串转为字节 

str.encode(str)

# bytes to str 字节转为字符串 

bytes.decode(bytes)
