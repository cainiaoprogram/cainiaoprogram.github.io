神经网络实现sklearn中鸢尾花数据集的分类

#导包
from sklearn import datasets
import numpy as np
from sklearn.model_selection import train_test_split

#处理数据集并且进行分类
iris_datas = datasets.load_iris()
X = iris_datas.data
y = iris_datas.target
# 将x数据划分在（0,1）区间内
# print(np.max(X))
# print(np.min(X))
X = X/8 *0.999+0.001
# 对数据集进行划分
X_train,X_test,y_train,y_test = train_test_split(X,y,test_size=0.2)
# print(X_train.shape)
# 对真实值的区间进行修改
ones = np.identity(3) # 定义一个三维的单位阵
for i in range(ones.shape[0]):
    for j in range(ones.shape[1]):
        if(ones[i,j]==1):
            ones[i,j]=0.999
        else:
            ones[i,j]=0.001
y_true = ones[y_train]


# 定义sigmoid函数
def sigmoid(x):
    return  1/(1+np.exp(-x))

# 计算准确率
def accuasy(y_test,rs):
    return sum(y_test == rs)/len(y_test)

# 设置神经网络，将隐藏层设置为1层，即有两个w参数矩阵，隐藏层数目为10个
W1 = np.random.normal(0.0,1,(4,10))
W2 = np.random.normal(0.0,1,(10,3))
# 定义学习率
learn_rate = 0.001
# 开始训练
for i in range(10000):
    y1 = np.dot(X_train,W1)
    y1_ = sigmoid(y1)
    y2 = np.dot(y1_,W2)
    y2_ = sigmoid(y2)
    error = 1/2*(y_true-y2_)**2 # 误差
    error_ = y2_-y_true  # 误差的梯度
    h_error = np.dot(error_,W2.T) # 隐藏层的误差梯度
    W2_ = np.dot(y1_.T,error_*y2_*(1-y2_)) # 求w2的梯度参数
    W1_ = np.dot(X_train.T,h_error*y1_*(1-y1_)) # 求w1的梯度参数
    W2 -=W2_ *learn_rate  #进行参数更新
    W1 -=W1_ *learn_rate


# 训练出的w1，w2参数矩阵带入到测试集进行计算
o1 = np.dot(X_test,W1)
a1 = sigmoid(o1)
o2 = np.dot(a1,W2)
a2 = sigmoid(o2)
# 计算出的测试值与真实值进行比较，计算出准确率
#  取a2中每一行最大值的索引与测试集进行比较
a=np.argmax(a2,axis = 1)
print(accuasy(y_test,a))

结果为：0.9333333333333333
