Docker配置Ubuntu容器、ssh远程连接该容器

# 详细请参考: https://juejin.cn/post/6982419819211522079
# 1. 拉取 ubuntu 镜像
docker pull ubuntu
# 2.查看是否拉取成功
dokcer images
# 3.运行容器
docker run --name 新建容器名称 -ti -d -p 3316:22 ubuntu 
# --name 指定生成的容器名称
# -i: 交互式模式运行 常与 -t配合使用
# -t: 为容器分配 tty 终端
# -d: 后台运行容器
# -p: 指定端口映射 一个指定端口 只能绑定一个容器 支持格式
# ubuntu 是镜像名称
# 4.查看容器是否运行成功
docker ps -a # 查看所有容器，包含启动和挂起的容器
# 注意 这个容器 是和极其干净的容器 所有的工具必须 自己安装
# 首先 apt-get update 升级系统  安装 apt-get install wget
# 正确退出方式 先按，ctrl+p；再按，ctrl+q。绝对不能使用exit或者ctrl+d来退
#出，这样整个系统就退出了。具体到MAC电脑应该使用control代替ctrl键
# 退出后在进入 ubuntu
# 1.首先用docker ps -a 查找到该CONTAINER ID对应编号（比如：0a3309a3b29e）
# 2.进入该系统，使用命令：docker attach 0a3309a3b29e

# 安装SSH 服务
# 1.进入终端容器
#  docker exec -i -t 新建容器名称 /bin/bash
# 2. 执行更新操作
#  apt-get update 
# 3. 安装 ssh-client、ssh-server
apt install openssh-client
apt install openssh-server
# 4.重启服务
#  /etc/init.d/ssh start
# 5. 查看是否成功
ps -ef | grep ssh
# 6.编辑sshd_config 文件
# 最后一行 添加 PermitRootLogin yes (这个要特别注意不然远程登录不了)
# 如果文件中 有 PermitRootLogin 配置 则修改 该配置为yes 即可
# 7. 重启SSH 服务
service ssh restart
# 8.设置 ssh 密码
passwd root # 密码会输入两遍
# 9. 查看容器IP 地址
apt-get install net-tools
# 10.保存修改的镜像
docker commit [容器id]
# 使用 ssh 工具连接注意 端口号

# 查看 ubuntu 版本
cat /etc/issure

# ssh 相关命令
service ssh start # 启动
service ssh stop  # 停止
service ssh restart # 重启
# 设置为开启自启服务
sudo systemctl enable ssh
