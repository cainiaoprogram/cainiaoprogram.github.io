C语言 字母小写转换大写

#include <stdio.h>

int main()
{
        char ch;
        scanf("%c", &ch);
        if (ch>='a' && ch<='z')
        {
            ch=ch-32;
        printf ("%c\n",ch);
        }
        else
        printf ("%c\n",ch);
        return 0;
}
