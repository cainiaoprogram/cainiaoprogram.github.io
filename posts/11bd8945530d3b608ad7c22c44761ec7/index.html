ScgcxxApplication.java

package com.johe.scgcxx;

import org.mybatis.spring.annotation.MapperScan;
import org.springframework.boot.SpringApplication;
import org.springframework.boot.autoconfigure.SpringBootApplication;
import org.springframework.web.servlet.config.annotation.ViewControllerRegistry;
import org.springframework.web.servlet.config.annotation.WebMvcConfigurer;

@SpringBootApplication
@MapperScan("com.johe.scgcxx.dao")
public class ScgcxxApplication implements WebMvcConfigurer {
    //实现WebMvcConfigurer 即 为了设置默认访问路径 /scgcxx
	@Override
	public void addViewControllers(ViewControllerRegistry registry) {
		registry.addViewController("/scgcxx").setViewName("redirect:/safty/login/toLogin");
	}

	public static void main(String[] args) {
		SpringApplication.run(ScgcxxApplication.class, args);
	}
}

SaftyLoginController.java

package com.johe.scgcxx.controller.safty;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Controller;
import org.springframework.web.bind.annotation.RequestMapping;

import com.johe.scgcxx.dto.Sys_User;
import com.johe.scgcxx.service.safty.SaftyLoginService;

//安全管理-用户登录
@Controller
public class SaftyLoginController {
	@Autowired
	private SaftyLoginService saftyLoginService;
	
	//   '/'表示绝对路径 不加 表示相对路径
	@RequestMapping("/safty/login/toLogin")
	public String toLoginJSP() {
		return "safty/login/SaftyLogin";
	}
	
	@RequestMapping("/safty/login/doLogin")
	public String toLogin(Sys_User user) {
		boolean flag = saftyLoginService.toLogin(user);
		if (flag) {
			System.out.println("登陆成功");
			return "redirect:/safty/home/toHome";
		}else {
			System.out.println("登录失败");
			return "redirect:/scgcxx";
		}
	}
	
	@RequestMapping("/safty/home/toHome")
	public String toHome() {
		return "safty/home/SaftyHome";
	}
}

SaftyLoginServiceImpl.java

package com.johe.scgcxx.service.safty.impl;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import com.johe.scgcxx.dao.safty.SaftyLoginDao;
import com.johe.scgcxx.dto.Sys_User;
import com.johe.scgcxx.service.safty.SaftyLoginService;

//安全管理-用户登录
@Service
@Transactional
public class SaftyLoginServiceImpl implements SaftyLoginService {
	
	@Autowired
	private SaftyLoginDao saftyLoginDao;
	
	@Override
	public boolean toLogin(Sys_User user) {
		com.johe.scgcxx.model.Sys_User breakUser = saftyLoginDao.getUser(user);
		if (breakUser != null) {
			return true;
		}else {
			return false;
		}
	}

}
