输入整数m，k将会出现大于m的k个与m最近的连续素数

#include<stdio.h>
int main()
{
 int m,k,xx[100],cnt=0,i;
 scanf("%d%d",&m,&k);
 while(k>cnt)
 {
  m++;
  for(i=2;i<=m;i++)//i<=m与i<m都可
  if(!(m%i))
  break;
  if(i==m)
  xx[cnt++]=m;//注意：先算出xx[0]，然后cnt++
 }
 for(i=0;i<cnt;i++)
 printf("%d  ",xx[i]);
 return 0;
}
