ios tableView那些事 (十一) 让 tableview 不可滚动或屏蔽掉

-(void)rightAction

{

     [firstViewsetHidden:NO];  

     self.tableView.scrollEnabled =NO;

}

-(void)leftAction

{

      [firstViewsetHidden:YES];  //让tableview上层的view 不隐藏

      self.tableView.scrollEnabled =YES;  //设置tableview 不能滚动

}
