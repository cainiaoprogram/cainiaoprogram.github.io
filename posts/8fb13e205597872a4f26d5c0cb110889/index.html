        public static bool CheckBracket(string inputStr)
        {
            //æ£€æŸ¥æ‹¬å·åŒ¹é…

            for (int i = 0; i < inputStr.Length; i++)
            {
                if (inputStr[i] == '(')
                {
                    leftBracket.Push(inputStr[i]);
                }
                else if (inputStr[i] == ')')
                {
                    if (leftBracket.Count == 0 || '(' != leftBracket.Pop())
                    {
                        return false;
                    }
                }
            }
            if (leftBracket.Count > 0)
            {
                return false;
            }
            return true;
        }

            for (int i = 0; i < inputStr.Length; i++)
            {
                if (inputStr[i] == '.'
                    || inputStr[i] == '+'
                    || inputStr[i] == '-'
                    || inputStr[i] == '*'
                    || inputStr[i] == '/'
                    || inputStr[i] == '('
                    || inputStr[i] == ')')
                    point.Push(inputStr[i]);
