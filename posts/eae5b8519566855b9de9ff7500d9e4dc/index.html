    <se:Name>gis_inner_702872535642697728</se:Name>

        <se:Rule>
          <se:Name></se:Name>
          <se:PolygonSymbolizer>

            </se:Stroke>
          </se:PolygonSymbolizer>
        </se:Rule>

          <se:TextSymbolizer>
            <se:Label>

              <ogc:Function name="strSubstring">

              <se:SvgParameter name="font-size">12</se:SvgParameter>

                  <se:AnchorPointX>0.5</se:AnchorPointX>

          </se:TextSymbolizer>
