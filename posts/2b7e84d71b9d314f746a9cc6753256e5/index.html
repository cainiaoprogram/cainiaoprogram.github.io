node.js jwt 封装 token生成 token验证

NodeJS 中如何应用 JWT

  1. 首先安装 jsonwebtoken
npm install jsonwebtoken --save
  1. 引入 jsonwebtoken
const jwt = require('jsonwebtoken')
  1. 封装 jwt.js
// token生成的密匙，根据自己需求定义
const jwtKey = 'dengxiansheng' 

// token生成函数(jwtSign)，有效时间为一个小时
const jwtSign = (data) => { 
  const token = jwt.sign(data, jwtKey, {expiresIn: 60 * 60})
  return token
}

// token验证函数(jwtCheck)
const jwtCheck = (req, res, next) => { 
  //前端headers传来的token值:
  const token = req.headers.token
  jwt.verify(token, jwtKey, (err, data) => {
    if (err) {
      res.send({
        status: '401',
        msg: 'token无效'
      })
    } else {
      req.jwtInfo = data
      next()
    }
  })
}

module.exports = {
  jwtSign,
  jwtCheck
}
  1. 登录接口获取token
//引入token生成函数
const {jwtSign} = require('../util/jwt')
// 用引入的jwtSign方法生成token并返回
const token = jwtSign({_mg_name: body.mg_name}) 
//token值传给前端
res.json({
	status: 200,
	message: '请求成功',
	token,
})
  1. 其他接口验证token
//引入token验证函数
const {jwtCheck} = require('../util/jwt')

// 用引入的jwtCheck方法验证token
router.get('/manager',jwtCheck, (req, res) => {
	res.send({
	    status: 200,
	     message: '验证token成功',
	 });
})
