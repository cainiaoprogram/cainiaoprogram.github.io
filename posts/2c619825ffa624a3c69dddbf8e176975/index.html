uniapp跳转微信小程序（5+app通用）

plus.share.getServices(function(res){
               var sweixin = null;  
               for(var i=0;i<res.length;i++){  
                 var t = res[i];  
                 if(t.id == 'weixin'){  
                       sweixin = t;  
                 }  
               }  
               if(sweixin){  
                 sweixin.launchMiniProgram({  
                    id: '',
                    type: 0,
                    path:'pages/user/wallet/xunyuji-recharge'//这里是指定页的路径，如需传参直接字符串拼接（首页可以省略）
                 });  
               }  
              },function(res){  
                  console.log(JSON.stringify(res));  
             });

需要用到share模块，去申请微信开放平台，然后在manifest.json里配置微信分享的相关参数
