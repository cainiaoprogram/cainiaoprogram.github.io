C语言--求100-999的水仙花数

//输出100-999的水仙花数

#include <stdio.h>

int main()
{
int i ;
int a ,b,c;
for(i= 100;i <= 999; i++ )
    {
        a = i%10; //个位
        b = i /10 %10; //十位
        c = i/100; //百位
        if (i == a*a*a + b*b*b + c*c*c)
        printf("i = %d\n",i);
    }
  
    

    return 0;
}
