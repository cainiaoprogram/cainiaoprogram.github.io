        static System.Collections.Concurrent.ConcurrentQueue<string> resultQueue = new System.Collections.Concurrent.ConcurrentQueue<string>();

            for (var k = 0; k < 10; k++)
            {
                var a1 = string.Format(urlPrefix, k);
                taskQueue.Enqueue(a1);
                for (var i = 0; i < 50; i++)
                {
                    var a2 = string.Format(urlPrefix, k + "." + i); 
                    taskQueue.Enqueue(a2);
                    for (var j = 0; j < 50; j++)
                    {
                        var a3 = string.Format(urlPrefix, k + "." + i + "." + j);
                        taskQueue.Enqueue(a3);
                    }
                }
            }
            var list = taskQueue.ToArray();
            foreach (var item in list)
            {
                System.Diagnostics.Debug.WriteLine(item);
            }
            Console.WriteLine("total:" + taskQueue.Count);
            var taskList = new System.Collections.Generic.List<Task>();
            for (int i = 0; i < 10; i++)
            {
                var task = Task.Factory.StartNew(DoTestUrl);
                taskList.Add(task);

            Task.WaitAll(taskList.ToArray());
            Console.WriteLine("sorted list:");
            var sortedList = resultQueue.ToArray().ToList();
            sortedList.Sort((x, y) =>
            {
                Regex r = new Regex("(\\d+\\.\\d+\\.\\d+\\.\\d+)|(\\d+\\.\\d+\\.\\d+)");
                var v1 = Version.Parse(r.Match(x).Value);
                var v2 = Version.Parse(r.Match(y).Value);
                return v1.CompareTo(v2);
            });
            foreach (var item in sortedList)
            {
                Console.WriteLine(item);
            }
            Console.WriteLine("finished");
            Console.ReadLine();
        }

        {
            while (!taskQueue.IsEmpty)
            {
                string url;

                        r.Method = "HEAD";
                        System.Net.HttpWebResponse s = r.GetResponse() as System.Net.HttpWebResponse;
                        if (s.StatusCode == System.Net.HttpStatusCode.OK)
                        {
                            Console.WriteLine(url);
                            resultQueue.Enqueue(url);
                        }
                        s.Close();
                    }
                    catch
                    { }
                }
            }
            
        }
    }
