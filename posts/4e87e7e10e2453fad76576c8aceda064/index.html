MATLAB中用fprintf函数实现矩阵原样输出

[l1,l2,l3] = size(A3);
fid = fopen(‘test.txt’,‘w’);
for j = 1:l3
for k = 1:l2
fprintf(fid,’%e\t’,A3(:,k,j));
fprintf(fid,’%s\n’,‘一行finish’);
end
fprintf(fid,’%s\n’,"------------------------------------------");
end
fclose(fid);
