PostgreSQL 循环遍历结果集

DO $$ 
DECLARE
  tmp RECORD;
BEGIN     FOR tmp IN SELECT id_,code_,name_ FROM t_ex_device ORDER BY id_ ASC LOOP 
          UPDATE t_ex_device SET name_=chr(int4(random()*26)+65) WHERE id_=tmp.id_;
    END LOOP;
END;        
$$;
