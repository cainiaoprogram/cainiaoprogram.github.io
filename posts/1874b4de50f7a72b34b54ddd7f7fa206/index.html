module fulladder(a,b,ci,sum,co);
    input [7:0]a,b;
    input ci;
    output [7:0] sum;
    output co;
    assign {co,sum}=a+b+ci;
endmodule

module fuadder_tb;
    reg[7:0] a,b;	//输入信号为reg型
    reg ci;	  
    wire[7:0] sum;
    wire co;		//输出信号为wire型
    fulladder U1(a,b,ci,sum,co);	//实例化
    initial
        begin
            a=8'b0000_0001, a=8'b0000_0001,ci=0;

        end
endmodule
