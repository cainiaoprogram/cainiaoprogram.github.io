#include<stdio.h>
#include<stdlib.h>

#define MaxSize 5
typedef int ElemType;
typedef struct Queue{
	ElemType data[MaxSize];// 数组，存储 MaxSize-1 个元素
	int front, rear;// 队列头 队列尾
}SqQueue;

void InitQueue(SqQueue& Q) {
	// 可以连写
	Q.front = Q.rear = 0;
	
}
bool isEmpty(SqQueue Q) {
	if (Q.front == Q.rear) {
		return true;
	}
	return false;
}
// 入队
bool EnQueue(SqQueue& Q,ElemType x) {
	if ((Q.rear + 1) % MaxSize == Q.front) {
		return false;// 队满
	}
	Q.data[Q.rear++] = x;
	Q.rear = Q.rear % MaxSize;
	return true;
}
// 出队
bool DeQueue(SqQueue& Q,ElemType& e) {
	if (isEmpty(Q)) {// 队空
		return false;
	}
	e = Q.data[Q.front];
	Q.front = (Q.front + 1) % MaxSize;
	return true;
}

// 打印队列
void PrintQueue(SqQueue Q) {
	if (isEmpty(Q)) {
		printf("队空\n");
	}
	for (int i = Q.front; i < Q.rear; i++) {
		printf(" %d ",Q.data[i]);
	}
	printf("\n");
}

int main() {
	// 定义一个队列，且初始化
	SqQueue Q;
	InitQueue(Q);
	// 入队测试
	EnQueue(Q,2);
	EnQueue(Q,3);
	EnQueue(Q,4);
	EnQueue(Q,5);
	PrintQueue(Q);
	// 出队测试
	ElemType temp;
	DeQueue(Q,temp);
	PrintQueue(Q);
	printf(" %d \n",temp);

	return 0;
}
