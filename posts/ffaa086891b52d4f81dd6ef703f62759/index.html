get请求参数字符串转json

 var json = {};
  var url = 'https://www.baidu.com/index.html?a=12&b=asdf&c=&d';
  url = url.replace(/.*\?/,'');
  var arr = url.split('&');
  var temp;
  for (var i = 0;i < arr.length;i++) {
    temp = arr[i].split('=');
    json[temp[0]] = temp[1];
  };
  console.log(json);
