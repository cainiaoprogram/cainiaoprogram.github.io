记录一下函数的一些内容：最大公约数和最小公倍数

# 最大公约数从最大的开始尝试
def big(x,y):
    (x, y) = (y, x) if x > y else (x, y)
    for i in range(x,1,-1):
        if x%i ==0 and y%i ==0:
            return i

# 最小公倍数从最大的开始到两数之积尝试
def mal(x,y):
    (x, y) = (y, x) if x > y else (x, y)
    for i in range(y, x*y):
        if i % x == 0 and i % y == 0:
            return i

#简便：最小公倍数 = 两数乘积 除以 最大公倍数  （涨知识了）
def mal2(x,y):
    return x*y // big(x,y)
