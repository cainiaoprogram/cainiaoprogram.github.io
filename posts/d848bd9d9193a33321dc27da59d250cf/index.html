html5 点一个按钮,播放一个相应的视频

<!doctype html>
<html>
<head>
<meta charset="utf-8">
<title>HTML5视频播放器</title>
</head>
<body>

<video id="myVideo">你的浏览器不支持video标签</video>
<a href="#" onClick="playlist(this)" id="https://www.vvuxx.com/v/21/">街霸</a>
<a href="#" onClick="playlist(this)" id="https://www.vvuxx.com/v/21/">铁拳</a>

<script>
function playlist(a){
    var MV=document.getElementById("myVideo");
    MV.src=a.id+'.mp4';
    MV.play();
    }
</script>
</body>
</html>
