第九题：第几个幸运数

'''其他做法很多，但是比较繁琐还需要大内存，这里我们使用暴力做法，
得到礼品的情况他357都是因数，所以直接利用这点 += 造到59084709587505'''
count=0
for i in range(50):
    for j in range(50):
        for k in range(50):
            r3=3**i
            r5=5**j
            r7=7**k
            if r3*r5*r7 < 59084709587505:
                count+=1
            else:
                break
print(count)

第十题：分解质因数（感觉这题比较难理解）

'''python分解质因数'''

def fj(num):
    print( str(num) +'=',end='')
    flag=0 #定义哨兵flag
    ls = []

    while True:
        if flag==1:
            break
        for i in range(2,num+1):
            if num%i==0:
                ls.append(str(i))
                if num==i:
                    flag=1
                    print('*'.join(ls))
                    break #单独上面这段相当于找出因数
                num=int(num/i) #注意前面for里num+1改变
                break
            
num = input('输入两个整数a，b:')
a,b = map(int,num.split())
n = a
while True:
    fj(n)
    if n == b:
        break
    n += 1

第十题第二种做法：看到更简便的做法，我加以注释解释了一下，来自寻找永不遗憾_CSDN博客

'''做这种题建议一定要做草稿本/画图！！！'''

def fj(num):
    for i in range(2, int(num**0.5)+1):
        # 最大质因数只到该数开方（固定的定理）
        if num % i == 0:
            # 找出该数开方前的所有因数
            print(('{}*').format(i), end='')
            # 成功找出因数后，num除以该因数再遍历(直接成功减少运算量!)
            fj(num // i)
            return
    # 下面那句用来print最后一个质因数或者类似17 19这种数字
    print(num, end='')
 
a, b = map(int, input().split())
for n in range(a, b+1):
    print('{}='.format(n), end='')
    fj(n)
    print()
