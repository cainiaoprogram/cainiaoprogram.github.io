java计算税收

题 :  国家推出新的个人所得税政策，起征点上调值5000元。也就是说税前工资扣除三险一金（三险一金数额假设是税前工资的10%）后如果不足5000元，则不交税。如果大于5000元，那么大于5000元的部分按梯度交税，具体梯度比例如下：

​    0 ~ 3000元的部分，交税3%            

​    3000 ~ 12000元的部分，交税10%

​    12000 ~ 25000的部分 ， 交税20%        

​    25000 ~ 35000的部分，交税25%

​    35000 ~ 55000的部分，交税30%        

​    55000 ~ 80000的部分，交税35%

​    超过80000的部分，交税45%

解:

class HomeWork6 {
    public static void main(String[] args) {
        //  收入 5000 以上考虑收税     税收 : 计算公式       收入 - 收入 * 0.1 (保险钱)   然后按档位缴费

        //  0 ~ 3000元的部分，交税3%             3000
        //
        // 3000 ~ 12000元的部分，交税10%      9000
        //
        // 12000 ~ 25000的部分 ， 交税20%    13000
        //
        // 25000 ~ 35000的部分，交税25%    10000
        //
        // 35000 ~ 55000的部分，交税30%     20000
        //
        // 55000 ~ 80000的部分，交税35%  // 25000
        //
        // 超过80000的部分，交税45%
        Scanner sr = new Scanner(System.in);
        System.out.println("请输入你的工资");
        double money = sr.nextDouble();
        //税收部分
        double shui = 0;
        //预处理
        double baoXian = money * 0.1;
        // 扣除保险之后的工资
        money = money - baoXian;
        //交税部分 多余 5000的
        double money1 = money - 5000;
        //税收
        if (money <= 5000) {
            System.out.println("您的工资收入为" + money);
        } else if (money1 >= 0 && money1 <= 3000) {
             shui = money1 * 0.03;
            money = money - shui;
            System.out.println(money);

        } else if (money1 > 3000 && money1 <= 12000) {
             shui = 3000 * 0.03 + (money1 - 3000) * 0.1;
            money = money - shui;
            System.out.println(money);
        } else if (money1 > 12000 && money1 <= 25000) {
             shui = 3000 * 0.03 + 9000 * 0.1 + (money1 - 12000) * 0.2;
            money = money - shui;
            System.out.println(money);

        } else if (money1 > 25000 && money1 <= 35000) {
             shui = 3000 * 0.3 + 9000 * 0.1 + 13000 * 0.2 + (money1 - 25000) * 0.25;
            money = money - shui;
            System.out.println(money);

        } else if (money1 > 35000 && money1 <= 55000) {
             shui = 3000 * 0.3 + 9000 * 0.1 + 13000 * 0.2 + 10000 * 0.25 + (money1 - 35000) * 0.3;
            money = money - shui;
            System.out.println(money);
        } else if (money1 > 55000 && money1 <= 80000) {
             shui = 3000 * 0.3 + 9000 * 0.1 + 13000 * 0.2 + 10000 * 0.25 +  20000 * 0.3 + (money1 - 55000) * 0.35;
            money = money - shui;
            System.out.println(money);
        } else if (money1 > 80000) {
             shui = 3000 * 0.3 + 9000 * 0.1 + 13000 * 0.2 + 10000 * 0.25 +  20000 * 0.3 + 25000 * 0.35 + (money1 - 80000) * 0.45;
            money = money - shui;
            System.out.println(money);
        }



    }

}
