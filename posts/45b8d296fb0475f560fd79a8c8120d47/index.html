【c++】复数加减法类

自己编写的一个实现复数加减法类，主要练练运算符重载
#include  < iostream.h >

class  complex
 ... {
public:
    complex(double xx = 0.0, double yy = 0.0)...{x = xx;y = yy;}
    complex operator+(complex);
    complex operator-(complex);
    void display();
private:
    double x;
    double y;
} ;

complex complex:: operator + (complex b)
 ... {
    complex temp;
    temp.x = x+b.x;
    temp.y = y+b.y;
    return temp;
}

complex complex:: operator - (complex b)
 ... {
    complex temp;
    temp.x = x-b.x;
    temp.y = y-b.y;
    return temp;
}

void  complex::display()
 ... {
    cout<<"("<<x<<")"<<"+"<<"("<<y<<")"<<"j"<<endl;
}

 ... {
    complex a(1.1, 1.2);
    complex b(2.1, 2.2);
    complex c;
    c = a+b;
    cout<<"a = ";
    a.display();
    cout<<"b = ";
    b.display();
    cout<<"c = a+b =";
    c.display();

    cout<<"c = a-b =";
    c.display();

}
