实现美元与人民币的转换
2022.4.16 ：1美元=6.37人民币

moneyStr = input('请输入带有标志(RMB,rmb,USD,usd)的钱数:')
if moneyStr[-3:] in ['RMB','rmb']:
    dollar = eval(moneyStr[0:-3])/6.37
    print('转换后的美元是%f USD'%(dollar))
elif moneyStr[-3:] in ['USD','usd']:
    yuan = 6.37*eval(moneyStr[0:-3])
    print('转换后的人民币是%f RMB'%yuan)
else:
    print('格式错误')
