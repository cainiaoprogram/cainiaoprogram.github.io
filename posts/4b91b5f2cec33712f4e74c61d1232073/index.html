西电期末1015.等差数列

一.题目

二.分析与思路

既然要求输出数据排序后在原数列中的位置，那最好使用结构体，排序后找出相邻差值的最大值和最小值，如果相等，那就是等差数列，反之则不是，按照题目操作即可。

三.代码实现

#include<bits/stdc++.h>//万能头
typedef struct number{
    int value;
    int id;
}num;//定义数列结构体
void swap(num* a,num* b){
    num t=*a;
    *a=*b;
    *b=t;
}//交换函数
void sort(num* a,int n){
    for(int i=0;i<n-1;i++){
        for(int j=0;j<n-1-i;j++){
            if(a[j+1].value<a[j].value)
            swap(&a[j],&a[j+1]);
        }
    }
}//排序函数
int main()
{
    int n;
    scanf("%d",&n);
    num a[n];
    for(int i=0;i<n;i++){
        scanf("%d",&a[i].value);
        a[i].id=i+1;//原数列位置
    }
    sort(a,n);//排序结构体
    int max=-1000000;
    int min=1000000;
    for(int i=0;i<n-1;i++){
        int d=a[i+1].value-a[i].value;
        if(d>max)max=d;
        if(d<min)min=d;
    }//打擂台求出排序后最大最小差值
    if(max==min){
        for(int i=0;i<n;i++){
            printf("%d ",a[i].id);
        }
    }//相等说明是等差
    else{
        printf("%d %d",max,min);
    }
    return 0;
}

四.评价

结构体排序+打擂台
