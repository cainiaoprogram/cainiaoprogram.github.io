2,创建service实现类HelloServiceImpl：


import com.ming.service.HelloService;
import org.apache.dubbo.config.annotation.Service;

@Service
public class HelloServiceImpl implements HelloService {
    @Override
    public String sayHello(String name) {
        return "hello：" + name;
    }
}

3，创建启动类DubboPureMain : 



import org.apache.dubbo.config.RegistryConfig;
import org.apache.dubbo.config.spring.context.annotation.EnableDubbo;
import org.springframework.context.annotation.AnnotationConfigApplicationContext;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.context.annotation.PropertySource;

import java.io.IOException;


public class DubboPureMain {
    public static void main(String[] args) throws IOException {
        AnnotationConfigApplicationContext context = new AnnotationConfigApplicationContext(ProviderConfiguration.class);
        context.start();
        System.in.read();

    }

    @Configuration
    @EnableDubbo(scanBasePackages = "com.ming.service.impl")
    @PropertySource("classpath:/dubbo-provider.properties")
    static class  ProviderConfiguration{
        @Bean
        public RegistryConfig registryConfig(){
            RegistryConfig registryConfig = new RegistryConfig();
            registryConfig.setAddress("zookeeper://127.0.0.1:2181");
            return registryConfig;
        }
    }
}

4,添加配置文件 dubbo-provider.properties：

dubbo.application.name=service-provider
dubbo.protocol.name=dubbo
dubbo.protocol.port=20881

5，启动zookeeper，启动启动类。

4，创建客户端module：service-consumer

1，引入jar包，和服务提供方一样

2，编写comsumerComponet:



import com.ming.service.HelloService;
import org.apache.dubbo.config.annotation.Reference;
import org.springframework.stereotype.Component;

@Component
public class ComsumerComponet {

    @Reference
    private HelloService helloService;

    public String sayHello(String name){
        return helloService.sayHello(name);
    }
}

3, 编写启动类：

package com.ming;

import com.ming.bean.ComsumerComponet;
import org.apache.dubbo.config.spring.context.annotation.EnableDubbo;
import org.springframework.context.annotation.AnnotationConfigApplicationContext;
import org.springframework.context.annotation.ComponentScan;
import org.springframework.context.annotation.Configuration;
import org.springframework.context.annotation.PropertySource;

import java.io.IOException;
import java.util.function.Consumer;

public class AnnotationConsumerMain {


    public static void main(String[] args) throws IOException {
        AnnotationConfigApplicationContext context = new AnnotationConfigApplicationContext(ConsumerConfigutation.class);
        context.start();
        ComsumerComponet comsumerComponet = context.getBean(ComsumerComponet.class);
        while (true){
            System.in.read();
            String heel = comsumerComponet.sayHello("IU");
            System.out.println("result: " + heel);
        }
    }

    @Configuration
    @PropertySource("classpath:/dubbo-consumer.properties")
    @ComponentScan(basePackages = "com.ming.bean")
    @EnableDubbo
    static class ConsumerConfigutation{

    }
}

4，编写配置文件：

dubbo.application.name=service-consumer
dubbo.registry.address=zookeeper://127.0.0.1:2181
