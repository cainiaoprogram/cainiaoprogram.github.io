1，lotus interopnet 安装

  • 终端代理（使用木马）
# export ALL_PROXY=socks5://127.0.0.1:1080
# git config --global http.proxy 'socks5://127.0.0.1:1080'
# export GO111MODULE=on
# export GOPROXY=https://goproxy.cn

# mkdir interopnet
# cd interopnet/

git clone --branch interopnet https://github.com/filecoin-project/lotus.git

# git show
commit 1bfa2311d6931dc0c961d42f561ba0268dbd21c3 (HEAD -> interopnet, tag: interop.4.24, origin/interopnet)
Author: Travis Person <travisperson@users.noreply.github.com>
Date:   Fri Apr 24 22:28:33 2020 +0000

    New interopt network info

make clean all && make install

# lotus --version
lotus version 0.3.0+git1bfa2311

2，运行

  • 设置环境变量vim /root/.bashrc
export LOTUS_STORAGE_PATH=/storage/lotuswork/lotusstorage
export LOTUS_PATH=/storage/lotuswork/lotus
export WORKER_PATH=/storage/lotuswork/lotusworker

export TMPDIR=/storage/lotuswork/tmpdir

#v25
export FIL_PROOFS_PARAMETER_CACHE=/storage/filecoin-proof-parameters-v25
export IPFS_GATEWAY=https://proof-parameters.s3.cn-south-1.jdcloud-oss.com/ipfs/

#bench
export RUSTFLAGS="-C target-cpu=native -g"
export FFI_BUILD_FROM_SOURCE=1
export FIL_PROOFS_MAXIMIZE_CACHING=1

export RUST_LOG=info
  • 检测复制证明是否下载完全
lotus fetch-params --proving-params=32GiB

3，bench编译

make clean deps bench

./bench sealing --storage-dir=/storage/benchtmp --sector-size=32GiB
