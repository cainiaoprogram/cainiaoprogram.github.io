OpenCV人脸检测之狗头替换

import cv2
img = cv2.imread(r'C:\Users\Curry\Desktop\NBA.png')
gray = cv2.cvtColor(img, code=cv2.COLOR_BGR2GRAY)
path = r"C:/Users/Curry/Desktop/cascades/haarcascade_frontalface_default.xml"
face_dector = cv2.CascadeClassifier(path)
faces = face_dector.detectMultiScale(gray)
goutou = cv2.imread(r"C:\Users\Curry\Desktop\goutou.png")
for x, y, w, h in faces:
    goutou_2 = cv2.resize(goutou, (w, h))
    img[y:y+h, x:x+w] = goutou_2
cv2.imshow('goutou', img)
cv2.waitKey(0)
cv2.destroyAllWindows()
