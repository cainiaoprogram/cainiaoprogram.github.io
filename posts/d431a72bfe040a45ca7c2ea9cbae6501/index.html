		printArr(str2);

	private static void printArr(String str) {
		for(int i=0;i<str.length();i++) {
			char letter = str.charAt(i);
			System.out.print(letter);
		}
		System.out.println("");
	}
	public static String reverse(String s) {
		String a ="";
		for (int i = s.length() - 1;i>=0;i--) {
			a= a + s.charAt(i);
		}
		return a;
	}
}
