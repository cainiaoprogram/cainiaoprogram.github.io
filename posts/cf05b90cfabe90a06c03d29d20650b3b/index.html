<span style="font-size:18px;">import java.util.Scanner;
/**
 * Created by Administrator on 2015/01/18.
 */
public class Main{
    public static void main(String[] args) {
        Scanner sc=new Scanner(System.in);
         while (sc.hasNext()){
            int Marx1hang=sc.nextInt();
             int Marx1lie2hang=sc.nextInt();
             int Marx2lie=sc.nextInt();
             int[][] Marx1=new int[Marx1hang][Marx1lie2hang];
             int[][] Marx2=new int[Marx1lie2hang][Marx2lie];
             for(int i=0;i<Marx1hang;i++){
                 for(int j=0;j<Marx1lie2hang;j++){
                     Marx1[i][j]=sc.nextInt();
                 }
             }
             for(int i=0;i<Marx1lie2hang;i++){
                 for(int j=0;j<Marx2lie;j++){
                     Marx2[i][j]=sc.nextInt();
                 }
             }
             for(int i=0;i<Marx1.length;i++){
                 StringBuffer sb=new StringBuffer();
                 for(int j=0;j<Marx2[0].length;j++){
                     int sum=0;
                     for (int k=0;k<Marx1lie2hang;k++){
                         sum+=(Marx1[i][k]*Marx2[k][j]);
                     }
                     sb.append(sum+" ");
                 }
                 System.out.println(sb.substring(0,sb.length()-1));
             }

        }

    }
}
