输入n个整数，分别统计输出正数、负数的个数。

#include <stdio.h>
int main(){
    int n;
    int count1=0;
    int count2=0;
    while(scanf("%d",&n)!=EOF){
        if(n>0){
            count1++;
        }
        else{
            count2++;
        }
    }
    printf("positive:%d\n",count1);
    printf("negative:%d",count2);
    return 0;
}
