对象转成get请求形式 page=1&limit=1

const data = {
      page: this.data.page,
      limit: this.data.limit,
    }

转换方法：

const setQueryConfig = function (item) {
  let _str = ''
  for (const o in item) {
    if (item[o] !== -1) {
      _str += o + '=' + item[o] + '&'
    }
  }
  _str = _str.substring(0, _str.length - 1) // 末尾是&
  return _str
}
