4，编写函数，实现两个整数位参数，计算加减乘除

{
echo "$a + $b =" $[$a+$b]
echo "$a - $b =" $[$a-$b]
echo "$a * $b =" $[$a*$b]
echo "$a / $b =" $[$a/$b]
}
read -p"输入两个数:"a b

done
