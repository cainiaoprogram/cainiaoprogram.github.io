搬自https://blog.csdn.net/seven_north/article/details/87073473博客

原理就是把一个汉字替换为英文字符aa，长度为2个字符，我这边就判断长度大约30就提示长度超出限制

var jmz = {};
jmz.GetLength = function(str) {
    return str.replace(/[\u0391-\uFFE5]/g,"aa").length;   //先把中文替换成两个字节的英文，在计算长度
};
