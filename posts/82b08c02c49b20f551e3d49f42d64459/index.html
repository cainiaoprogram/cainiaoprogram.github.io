第62篇：批量去除EXCEL文件密码

Sub 清除密码()
Excel.Application.ScreenUpdating = False
arr = Sheet1.UsedRange.Value
For 行号 = 2 To UBound(arr)
文件 = "此处写文件路径 以\结尾" & arr(行号, 1) & ".xlsx"
Set w1 = Workbooks.Open(文件, Password:=arr(行号, 2))
w1.Password = ""
w1.Save
w1.Close
Next
Excel.Application.ScreenUpdating = True

End Sub
