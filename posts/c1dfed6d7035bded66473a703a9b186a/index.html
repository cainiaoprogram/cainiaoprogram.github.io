关于ts文件中，引入vue文件报红线的问题

问题就是ts不会识别vue，会认为没有这样的模块

找到你的env.d.ts中写入下面代码

declare module '*.vue' {
  import { DefineComponent } from 'vue'
  const component: DefineComponent<object, object, any>
  export default component
}
