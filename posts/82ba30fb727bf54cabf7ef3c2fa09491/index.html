redisTemplate 根据key前缀批量删除

/**
 * 根据key前缀批量删除
 *
 * @param keyPrefix 键前缀字符串
 * @return 结果
 */
public boolean delByPrefix(String keyPrefix) {
    if (keyPrefix != null) {
        Set<String> keys = redisTemplate.keys(Pattern.matches("\\*$", keyPrefix) ? keyPrefix : keyPrefix + "*");
        redisTemplate.delete(keys);
        return true;
    }
    return false;
}
