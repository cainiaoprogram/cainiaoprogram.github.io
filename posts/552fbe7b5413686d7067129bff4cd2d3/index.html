2018 年 数据证书的生成

keytool -genkey -alias zljh  -keypass talent  -keyalg RSA -keysize 1024 -validity 3650 -keystore /home/certificates/zljh.keystore -storepass talent


keytool -genkey -alias client1 -keypass 123456 -keyalg RSA -keysize 1024  -validity 15 -storetype PKCS12 -keystore /home/certificates/client1.p12 -storepass 123456




keytool -export -alias client -keystore /home/certificates/client1.p12 -storetype PKCS12 -keypass 123456 -file /home/certificates/client.cer



keytool  -import  -v  -file  /home/certificates/client1.cer   -keystore  /home/certificates/zljh.keystore  -storepass  123456


keytool -keystore  /home/certificates/zljh.keystore  -export -alias zljh   -file /home/certificates/client1.cer
