 代码

a = int(input('输入第一个数：'))
b = int(input('输入第二个数：'))
num1 = a + b
num2 = a - b
num3 = a * b
num4 = a / b
print('{} + {} = {}'.format(a, b, num1))
print('{} - {} = {}'.format(a, b, num2))
print('{} * {} = {}'.format(a, b, num3))
print('{} / {} = {}'.format(a, b, num4))
