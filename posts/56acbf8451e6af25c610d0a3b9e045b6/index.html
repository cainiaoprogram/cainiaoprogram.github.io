解题说明：三个数中两个数一样，找出另一个数，水题，直接比较即可。
 

#include<stdio.h>

int main(void)
{
	int t, a, b, c;
	scanf("%d", &t);
	while (t--)
	{
		scanf("%d%d%d", &a, &b, &c);
		printf("%d\n", a == b ? c : a == c ? b : a);
	}
	return 0;
}
