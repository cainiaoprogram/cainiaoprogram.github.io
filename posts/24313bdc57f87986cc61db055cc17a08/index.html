python-猜数字游戏（0-9）

from random import *
x=randint(0,9)#预设0-9的整数
n=0
print("请输入0-9的数字")
while(1):
    m=eval(input())
    if x<m:
        print("遗憾，太大了")
        n+=1
        continue
    if(x>m):
        print("遗憾，太小了")
        n+=1
        continue
    else:
        n+=1
        break
print("预测{:}次，你猜中了！".format(n))
