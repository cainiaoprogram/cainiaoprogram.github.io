LVM方式挂载硬盘

使用lvm方式挂载新硬盘
1）查看设备详情

[root@localhost ~]# fdisk -l | grep dev
Disk /dev/sda: 42.9 GB, 42949672960 bytes, 83886080 sectors
/dev/sda1   *        2048     2099199     1048576   83  Linux
/dev/sda2         2099200    83886079    40893440   8e  Linux LVM
Disk /dev/mapper/centos-root: 37.6 GB, 37576769536 bytes, 73392128 sectors
Disk /dev/mapper/centos-swap: 4294 MB, 4294967296 bytes, 8388608 sectors
Disk /dev/sdb: 214.7 GB, 214748364800 bytes, 419430400 sectors

上面的/dev/sdb是新添加的硬盘

2）创建pv

[root@localhost ~]# pvcreate /dev/sdb
  Physical volume "/dev/sdb" successfully created.

3）创建vg

[root@localhost ~]# vgcreate locvg /dev/sdb
  Volume group "locvg" successfully created

[root@localhost ~]# vgs
  VG     #PV #LV #SN Attr   VSize    VFree   
  centos   1   2   0 wz--n-  <39.00g       0 
  locvg    1   0   0 wz--n- <200.00g <200.00g

4）创建lv

[root@localhost ~]# lvcreate -l 100%VG -n lvgfs locvg
  Logical volume "lvgfs" created.

[root@localhost ~]# lvs
  LV    VG     Attr       LSize    Pool Origin Data%  Meta%  Move Log Cpy%Sync Convert
  root  centos -wi-ao----  <35.00g                                                    
  swap  centos -wi-ao----    4.00g                                                    
  lvgfs locvg  -wi-a----- <200.00g

4）格式化

[root@localhost ~]# mkfs.ext4 /dev/mapper/locvg-lvgfs

5）创建挂载点

[root@localhost ~]# mkdir /database

6）挂载设备

[root@localhost ~]# mount /dev/mapper/locvg-lvgfs /database/
[root@localhost ~]# df -h
/dev/mapper/locvg-lvgfs ext4      197G   61M  187G   1% /database

7）写入开机加载列表，开机自动识别

[root@localhost ~]# echo /dev/mapper/locvg-lvgfs /database ext4 defaults 0 0 >> /etc/fstab

8）卸载并进行重新挂载

[root@localhost ~]# umount /database/
[root@localhost ~]# mount /database/
[root@localhost ~]# df -hT
Filesystem              Type      Size  Used Avail Use% Mounted on
