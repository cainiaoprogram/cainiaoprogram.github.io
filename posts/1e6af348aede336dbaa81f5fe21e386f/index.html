c语言实现字符串的逆序输出

#include<stdio.h>
#include<string.h>
main()
{
  int i,j,t,n;
  char a[10];
  printf("请输入字符串:");
  gets(a);
  n=strlen(a);
  for(i=0;i<=n/2;i++)
  {
  	t=a[i];
  	a[i]=a[n-1-i];
  	a[n-1-i]=t;
  	
	  }	
	  for(j=0;j<n;j++)
	  printf("%c",a[j]);
	  printf("\n"); 
 }
