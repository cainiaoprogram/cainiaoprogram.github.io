用C语言输出平行四边形图案


程序分析：首先需要输出平行四边形前面的空格，可以将其看为一个等腰直角三角形进行输出（循环结束条件为行号等于列号）

#include<stdio.h>
#include<math.h>
#include<stdlib.h>
int main()
{
	int i, j;
	for (i = 0; i < 5 ; i++)
	{
		for (j = 0; j <= i; j++)
		{
			printf(" ");//打印前面的空格
		}
		printf("* * * * *\n");
	}
	system("pause");
	return 0;
}
